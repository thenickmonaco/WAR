5/// A type to represent 4-byte SFNT tags.
6///
7/// The easiest way to create a tag is by using its `From<&[u8; 4]>` impl:
8///
9/// ```
10/// # use harfbuzz_rs::Tag;
11/// let tag: Tag = b"abcd".into();
12/// assert_eq!(&tag.to_bytes(), b"abcd");
13/// ```
14///
15/// Tables, features, etc. in OpenType and many other font formats use SFNT tags
16/// as identifiers. These are 4-bytes long and usually each byte represents an
17/// ASCII value. `Tag` provides methods to create such identifiers from
18/// individual `chars` or a `str` slice and to get the string representation of
19/// a `Tag`.
20#[derive(Copy, Clone, Hash, PartialEq, Eq)]
21#[repr(transparent)]
22pub struct Tag(pub hb::hb_tag_t);
23
24impl Tag {
25    /// Create a `Tag` from its four-char textual representation.
26    ///
27    /// All the arguments must be ASCII values.
28    ///
29    /// # Examples
30    ///
31    /// ```
32    /// use harfbuzz_rs::Tag;
33    /// let cmap_tag = Tag::new('c', 'm', 'a', 'p');
34    /// assert_eq!(cmap_tag.to_string(), "cmap")
35    /// ```
36    ///
37    pub const fn new(a: char, b: char, c: char, d: char) -> Self {
38        Tag(((a as u32) << 24) | ((b as u32) << 16) | ((c as u32) << 8) | (d as u32))
39    }
40
41    fn tag_to_string(self) -> String {
42        let mut buf: [u8; 4] = [0; 4];
43        unsafe { hb::hb_tag_to_string(self.0, buf.as_mut_ptr() as *mut _) };
44        String::from_utf8_lossy(&buf).into()
45    }
46
47    /// Returns tag as 4-element byte array.
48    ///
49    /// # Examples
50    /// ```
51    /// # use harfbuzz_rs::Tag;
52    /// let tag = Tag::new('a', 'b', 'c', 'd');
53    /// assert_eq!(&tag.to_bytes(), b"abcd");
54    /// ```
55    pub const fn to_bytes(self) -> [u8; 4] {
56        #[allow(clippy::identity_op)]
57        [
58            (self.0 >> 24 & 0xff) as u8,
59            (self.0 >> 16 & 0xff) as u8,
60            (self.0 >> 8 & 0xff) as u8,
61            (self.0 >> 0 & 0xff) as u8,
62        ]
63    }
64}

68impl Debug for Tag {
69    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
70        let string = self.tag_to_string();
71        let mut chars = string.chars().chain(std::iter::repeat('\u{FFFD}'));
72        write!(
73            f,
74            "Tag({:?}, {:?}, {:?}, {:?})",
75            chars.next().unwrap(),
76            chars.next().unwrap(),
77            chars.next().unwrap(),
78            chars.next().unwrap()
79        )
80    }
81}

107/// An Error generated when a `Tag` fails to parse from a `&str` with the
108/// `from_str` function.
109pub enum TagFromStrErr {
110    /// The string contains non-ASCII characters.
111    NonAscii,
112    /// The string has length zero.
113    ZeroLengthString,
114}
115
116use std;
117use std::str::FromStr;
118
119impl FromStr for Tag {
120    type Err = TagFromStrErr;
121    /// Parses a `Tag` from a `&str` that contains four or less ASCII
122    /// characters. When the string's length is smaller than 4 it is extended
123    /// with `' '` (Space) characters. The remaining bytes of strings longer
124    /// than 4 bytes are ignored.
125    ///
126    /// # Examples
127    ///
128    /// ```
129    /// use harfbuzz_rs::Tag;
130    /// use std::str::FromStr;
131    /// let tag1 = Tag::from_str("ABCD").unwrap();
132    /// let tag2 = Tag::new('A', 'B', 'C', 'D');
133    /// assert_eq!(tag1, tag2);
134    /// ```
135    ///
136    fn from_str(s: &str) -> Result<Tag, TagFromStrErr> {
137        if !s.is_ascii() {
138            return Err(TagFromStrErr::NonAscii);
139        }
140        if s.is_empty() {
141            return Err(TagFromStrErr::ZeroLengthString);
142        }
143        let len = std::cmp::max(s.len(), 4) as i32;
144        unsafe { Ok(Tag(hb::hb_tag_from_string(s.as_ptr() as *mut _, len))) }
145    }
146}

148/// Defines the direction in which text is to be read.
149#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
150pub enum Direction {
151    /// Initial, unset direction.
152    Invalid,
153    /// Text is set horizontally from left to right.
154    Ltr,
155    /// Text is set horizontally from right to left.
156    Rtl,
157    /// Text is set vertically from top to bottom.
158    Ttb,
159    /// Text is set vertically from bottom to top.
160    Btt,
161}
162
163impl Direction {
164    /// Convert into raw value of type `hb_direction_t`.
165    pub fn to_raw(self) -> hb::hb_direction_t {
166        match self {
167            Direction::Invalid => hb::HB_DIRECTION_INVALID,
168            Direction::Ltr => hb::HB_DIRECTION_LTR,
169            Direction::Rtl => hb::HB_DIRECTION_RTL,
170            Direction::Ttb => hb::HB_DIRECTION_TTB,
171            Direction::Btt => hb::HB_DIRECTION_BTT,
172        }
173    }
174
175    /// Create from raw value of type `hb_direction_t`.
176    pub fn from_raw(dir: hb::hb_direction_t) -> Self {
177        match dir {
178            hb::HB_DIRECTION_LTR => Direction::Ltr,
179            hb::HB_DIRECTION_RTL => Direction::Rtl,
180            hb::HB_DIRECTION_TTB => Direction::Ttb,
181            hb::HB_DIRECTION_BTT => Direction::Btt,
182            _ => Direction::Invalid,
183        }
184    }
185}

203impl Display for Language {
204    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
205        let string = unsafe {
206            let char_ptr = hb::hb_language_to_string(self.0);
207            if char_ptr.is_null() {
208                return Err(fmt::Error);
209            }
210            CStr::from_ptr(char_ptr)
211                .to_str()
212                .expect("String representation of language is not valid utf8.")
213        };
214        write!(f, "{}", string)
215    }

219pub struct InvalidLanguage;
220
221impl FromStr for Language {
222    type Err = InvalidLanguage;
223    fn from_str(s: &str) -> Result<Language, InvalidLanguage> {
224        let len = std::cmp::min(s.len(), std::i32::MAX as _) as i32;
225        let lang = unsafe { hb::hb_language_from_string(s.as_ptr() as *mut _, len) };
226        if lang.is_null() {
227            Err(InvalidLanguage {})
228        } else {
229            Ok(Language(lang))
230        }
231    }

251/// A trait which is implemented for all harffbuzz wrapper structs. It exposes
252/// common functionality for converting from and to the underlying raw harfbuzz
253/// pointers that are useful for ffi.
254///
255/// # Safety
256///
257/// This trait may only be implemented for structs that are zero-sized and is
258/// therefore unsafe to implement.
259pub unsafe trait HarfbuzzObject: Sized {
260    /// Type of the raw harfbuzz object.
261    type Raw;
262
263    /// Creates a reference from a harfbuzz object pointer.
264    ///
265    /// Unsafe because a raw pointer may be accessed. The reference count is not
266    /// changed. Should not be called directly by a library user.
267    ///
268    /// Use the Owned and Shared abstractions instead.
269    #[doc(hidden)]
270    unsafe fn from_raw(val: *const Self::Raw) -> Self;
271
272    /// Returns the underlying harfbuzz object pointer.
273    ///
274    /// The caller must ensure, that this pointer is not used after `self`'s
275    /// destruction.
276    fn as_raw(&self) -> *mut Self::Raw;
277
278    /// Returns the underlying harfbuzz object pointer with the intention of
279    /// mutating the object.
280    ///
281    /// The caller must ensure, that this pointer is not used after `self`'s
282    /// destruction.
283    fn as_raw_mut(&mut self) -> *mut Self::Raw {
284        self.as_raw()
285    }
286
287    /// Increases the reference count of the HarfBuzz object.
288    ///
289    /// Wraps a `hb_TYPE_reference()` call.
290    ///
291    /// # Safety
292    ///
293    /// While no undefined behavior can be introduced only by increasing the
294    /// reference count (I think) this method is still marked unsafe since there
295    /// should be no need for it to be called from safe code.
296    unsafe fn reference(&self);
297
298    /// Decreases the reference count of the HarfBuzz object and destroys it if
299    /// the reference count reaches zero.
300    ///
301    /// Wraps a `hb_TYPE_destroy()` call.
302    ///
303    /// # Safety
304    ///
305    /// You always have to call `reference` first before using this method.
306    /// Otherwise you might accidentally hold on to already destroyed objects
307    /// and causing UB.
308    unsafe fn dereference(&self);
309}

311/// A smart pointer that wraps an atomically reference counted HarfBuzz object.
312///
313/// Usually you don't create a `Shared` yourself, but get it from another
314/// function in this crate. You can just use the methods of the wrapped object
315/// through its `Deref` implementation.
316///
317/// A `Shared` is a safe wrapper for reference counted HarfBuzz objects and
318/// provides shared immutable access to its inner object. As HarfBuzz' objects
319/// are all thread-safe `Shared` implements `Send` and `Sync`.
320///
321/// Tries to mirror the stdlib `Arc` interface where applicable as HarfBuzz'
322/// reference counting has similar semantics.
323#[derive(Debug, PartialEq, Eq)]
324pub struct Shared<T: HarfbuzzObject> {
325    object: T,
326}
327
328impl<T: HarfbuzzObject> Shared<T> {
329    /// Creates a `Shared` from an owned raw harfbuzz pointer.
330    ///
331    /// # Safety
332    ///
333    /// Transfers ownership. _Use of the original pointer is now forbidden!_
334    /// Unsafe because dereferencing a raw pointer is necessary.
335    pub unsafe fn from_raw_owned(raw: *mut T::Raw) -> Self {
336        let object = T::from_raw(raw);
337        Shared { object }
338    }
339
340    /// Converts `self` into the underlying harfbuzz object pointer value. The
341    /// resulting pointer has to be manually destroyed using `hb_TYPE_destroy`
342    /// or be converted back into the wrapper using the `from_raw` function to
343    /// avoid leaking memory.
344    pub fn into_raw(shared: Shared<T>) -> *mut T::Raw {
345        let result = shared.object.as_raw();
346        std::mem::forget(shared);
347        result
348    }
349
350    /// Creates a `Shared` by cloning a raw harfbuzz pointer.
351
352    ///
353    /// The original pointer can still be safely used but must be released at
354    /// the end to avoid memory leaks.
355    ///
356    /// # Safety
357    ///
358    /// `raw` must be a valid pointer to the corresponding harfbuzz object or
359    /// the behavior is undefined.
360    ///
361    /// Internally this method increases the reference count of `raw` so it is
362    /// safe to call `hb_destroy_[...]` on `raw` after using `from_raw_ref`.
363    pub unsafe fn from_raw_ref(raw: *mut T::Raw) -> Self {
364        let object = T::from_raw(raw);
365        object.reference();
366        Shared { object }
367    }
368}
369
370impl<T: HarfbuzzObject> Clone for Shared<T> {
371    /// Returns a copy and increases the reference count.
372    ///
373    /// This behaviour is exactly like `Arc::clone` in the standard library.
374    fn clone(&self) -> Self {
375        unsafe { Self::from_raw_ref(self.object.as_raw()) }
376    }
377}

380    type Target = T;
381
382    fn deref(&self) -> &T {
383        &self.object
384    }
385}

390    }

410/// A smart pointer that wraps a singly owned harfbuzz object.
411///
412/// A `Owned` is used to wrap freshly created owned HarfBuzz objects. It permits
413/// mutable, non-shared access to the enclosed HarfBuzz value so it can be used
414/// e.g. to set up a `Font` or `Face` after its creation.
415///
416/// There is no safe way to construct an `Owned` pointer and usually you don't
417/// need to create a `Owned` yourself, but get it from another function in this
418/// crate. You can just use the methods of the wrapped object through its
419/// `Deref` implementation.
420///
421/// Interaction with `Shared`
422/// -------------------------
423/// When you are finished mutating the inner value, you usually want to pass it
424/// to other HarfBuzz functions that expect shared access. Thus you need to
425/// convert the `Owned` to a `Shared` pointer using `.into()`. Note however that
426/// once a value is converted to  a `Shared<T>`, it will not possible to mutate
427/// it anymore.
428#[derive(Debug, PartialEq, Eq)]
429pub struct Owned<T: HarfbuzzObject> {
430    object: T,
431}
432
433impl<T: HarfbuzzObject> Owned<T> {
434    /// Creates a `Owned` safely wrapping a raw harfbuzz pointer.
435    ///
436    /// # Safety
437    ///
438    /// This fully transfers ownership. _Use of the original pointer is now
439    /// forbidden!_ Unsafe because a dereference of a raw pointer is necessary.
440    ///
441    /// Use this only to wrap freshly created HarfBuzz object that is not
442    /// shared! Otherwise it is possible to have aliasing mutable references
443    /// from safe code
444    pub unsafe fn from_raw(raw: *mut T::Raw) -> Self {
445        Owned {
446            object: T::from_raw(raw),
447        }
448    }
449
450    /// Converts `self` into the underlying harfbuzz object pointer value. The
451    /// resulting pointer has to be manually destroyed using `hb_TYPE_destroy`
452    /// or be converted back into the wrapper using the `from_raw` function to
453    /// avoid leaking memory.
454    pub fn into_raw(owned: Owned<T>) -> *mut T::Raw {
455        let result = owned.object.as_raw();
456        std::mem::forget(owned);
457        result
458    }
459
460    /// Demotes an `Owned` pointer to a `Shared` pointer.
461    ///
462    /// Use this method when you don't need exclusive (mutable) access to the
463    /// object anymore. For differences between `Owned` and `Shared` pointers
464    /// see the documentation on the respective structs.
465    ///
466    /// Note that `Shared<T>` also implements `From<Owned<T>>` which allows
467    /// implicit conversions in many functions.
468    #[allow(clippy::wrong_self_convention)] //< backward compatibility is more important than clippy
469    pub fn to_shared(self) -> Shared<T> {
470        self.into()
471    }
472}
473
474impl<T: HarfbuzzObject> Drop for Owned<T> {
475    fn drop(&mut self) {
476        unsafe { self.dereference() }
477    }
478}
479
480impl<T: HarfbuzzObject> Deref for Owned<T> {
481    type Target = T;
482
483    fn deref(&self) -> &T {
484        &self.object
485    }
486}
487
488impl<T: HarfbuzzObject> DerefMut for Owned<T> {
489    fn deref_mut(&mut self) -> &mut T {
490        &mut self.object
491    }
492}

526    #[derive(Debug, Clone)]
527    struct ReferenceCounter {
528        share_count: Rc<Cell<isize>>,
529    }
530
531    unsafe impl HarfbuzzObject for ReferenceCounter {
532        type Raw = Cell<isize>;
533
534        unsafe fn from_raw(raw: *const Cell<isize>) -> Self {
535            ReferenceCounter {
536                share_count: Rc::from_raw(raw as *mut _),
537            }
538        }
539
540        fn as_raw(&self) -> *mut Cell<isize> {
541            Rc::into_raw(self.share_count.clone()) as *mut _
542        }
543
544        unsafe fn reference(&self) {
545            println!("referencing {:?}", self);
546            let rc = self.share_count.get();
547            self.share_count.set(rc + 1);
548        }
549
550        unsafe fn dereference(&self) {
551            println!("dereferencing {:?}", self);
552            let rc = self.share_count.get();
553            self.share_count.set(rc - 1);
554        }
555    }

557    #[test]
558    fn reference_counting_shared() {
559        // Mimic a C-API that returns a pointer to a reference counted value.
560        let object = ReferenceCounter {
561            share_count: Rc::new(Cell::new(1)),
562        };
563
564        // this clones the underlying `Rc`
565        let raw = object.as_raw();
566
567        // so we expect two shared owners
568        assert_eq!(Rc::strong_count(&object.share_count), 2);
569
570        let shared: Shared<ReferenceCounter> = unsafe { Shared::from_raw_owned(raw) };
571
572        assert_eq!(shared.share_count.get(), 1);
573        {
574            // we create another `Shared` pointer...
575            let shared2 = Shared::clone(&shared);
576            // which clones
577            assert_eq!(shared.share_count.get(), 2);
578            mem::drop(shared2);
579        }
580        assert_eq!(shared.share_count.get(), 1);
581        mem::drop(shared);
582
583        assert_eq!(object.share_count.get(), 0);
584
585        // ensure there are no dangling references
586        assert_eq!(Rc::strong_count(&object.share_count), 1);
587    }
