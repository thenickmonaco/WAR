32/// Status of a Lua thread (or coroutine).
33#[derive(Debug, Copy, Clone, Eq, PartialEq)]
34pub enum ThreadStatus {
35    /// The thread was just created, or is suspended because it has called `coroutine.yield`.
36    ///
37    /// If a thread is in this state, it can be resumed by calling [`Thread::resume`].
38    ///
39    /// [`Thread::resume`]: crate::Thread::resume
40    Resumable,
41    /// Either the thread has finished executing, or the thread is currently running.
42    Unresumable,
43    /// The thread has raised a Lua error during execution.
44    Error,
45}

48#[derive(Clone, Debug)]

67impl<'lua> Thread<'lua> {
68    /// Resumes execution of this thread.
69    ///
70    /// Equivalent to `coroutine.resume`.
71    ///
72    /// Passes `args` as arguments to the thread. If the coroutine has called `coroutine.yield`, it
73    /// will return these arguments. Otherwise, the coroutine wasn't yet started, so the arguments
74    /// are passed to its main function.
75    ///
76    /// If the thread is no longer in `Active` state (meaning it has finished execution or
77    /// encountered an error), this will return `Err(CoroutineInactive)`, otherwise will return `Ok`
78    /// as follows:
79    ///
80    /// If the thread calls `coroutine.yield`, returns the values passed to `yield`. If the thread
81    /// `return`s values from its main function, returns those.
82    ///
83    /// # Examples
84    ///
85    /// ```
86    /// # use mlua::{Error, Lua, Result, Thread};
87    /// # fn main() -> Result<()> {
88    /// # let lua = Lua::new();
89    /// let thread: Thread = lua.load(r#"
90    ///     coroutine.create(function(arg)
91    ///         assert(arg == 42)
92    ///         local yieldarg = coroutine.yield(123)
93    ///         assert(yieldarg == 43)
94    ///         return 987
95    ///     end)
96    /// "#).eval()?;
97    ///
98    /// assert_eq!(thread.resume::<_, u32>(42)?, 123);
99    /// assert_eq!(thread.resume::<_, u32>(43)?, 987);
100    ///
101    /// // The coroutine has now returned, so `resume` will fail
102    /// match thread.resume::<_, u32>(()) {
103    ///     Err(Error::CoroutineInactive) => {},
104    ///     unexpected => panic!("unexpected result {:?}", unexpected),
105    /// }
106    /// # Ok(())
107    /// # }
108    /// ```
109    pub fn resume<A, R>(&self, args: A) -> Result<R>
110    where
111        A: ToLuaMulti<'lua>,
112        R: FromLuaMulti<'lua>,
113    {
114        let lua = self.0.lua;
115        let mut args = args.to_lua_multi(lua)?;
116        let nargs = args.len() as c_int;
117        let results = unsafe {
118            let _sg = StackGuard::new(lua.state);
119            check_stack(lua.state, cmp::max(nargs + 1, 3))?;
120
121            let thread_state = ffi::lua_tothread(lua.ref_thread(), self.0.index);
122
123            let status = ffi::lua_status(thread_state);
124            if status != ffi::LUA_YIELD && ffi::lua_gettop(thread_state) == 0 {
125                return Err(Error::CoroutineInactive);
126            }
127
128            check_stack(thread_state, nargs)?;
129            for arg in args.drain_all() {
130                lua.push_value(arg)?;
131            }
132            ffi::lua_xmove(lua.state, thread_state, nargs);
133
134            let mut nresults = 0;
135
136            let ret = ffi::lua_resume(thread_state, lua.state, nargs, &mut nresults as *mut c_int);
137            if ret != ffi::LUA_OK && ret != ffi::LUA_YIELD {
138                check_stack(lua.state, 3)?;
139                protect_lua!(lua.state, 0, 1, |state| error_traceback_thread(
140                    state,
141                    thread_state
142                ))?;
143                return Err(pop_error(lua.state, ret));
144            }
145
146            let mut results = args; // Reuse MultiValue container
147            check_stack(lua.state, nresults + 2)?; // 2 is extra for `lua.pop_value()` below
148            ffi::lua_xmove(thread_state, lua.state, nresults);
149
150            for _ in 0..nresults {
151                results.push_front(lua.pop_value());
152            }
153            results
154        };
155        R::from_lua_multi(results, lua)
156    }

158    /// Gets the status of the thread.
159    pub fn status(&self) -> ThreadStatus {
160        let lua = self.0.lua;
161        unsafe {
162            let thread_state = ffi::lua_tothread(lua.ref_thread(), self.0.index);
163
164            let status = ffi::lua_status(thread_state);
165            if status != ffi::LUA_OK && status != ffi::LUA_YIELD {
166                ThreadStatus::Error
167            } else if status == ffi::LUA_YIELD || ffi::lua_gettop(thread_state) > 0 {
168                ThreadStatus::Resumable
169            } else {
170                ThreadStatus::Unresumable
171            }
172        }
173    }

175    /// Resets a thread
176    ///
177    /// In [Lua 5.4]: cleans its call stack and closes all pending to-be-closed variables.
178    /// Returns a error in case of either the original error that stopped the thread or errors
179    /// in closing methods.
180    ///
181    /// In [LuaJIT] and Luau: resets to the initial state of a newly created Lua thread.
182    /// Lua threads in arbitrary states (like yielded or errored) can be reset properly.
183    ///
184    /// Sets a Lua function for the thread afterwards.
185    ///
186    /// Requires `feature = "lua54"` OR `feature = "luajit,vendored"` OR `feature = "luau"`
187    ///
188    /// [Lua 5.4]: https://www.lua.org/manual/5.4/manual.html#lua_resetthread
189    /// [LuaJIT]: https://github.com/openresty/luajit2#lua_resetthread
190    #[cfg(any(
191        feature = "lua54",
192        all(feature = "luajit", feature = "vendored"),
193        feature = "luau",
194    ))]
195    pub fn reset(&self, func: Function<'lua>) -> Result<()> {
196        let lua = self.0.lua;
197        unsafe {
198            let _sg = StackGuard::new(lua.state);
199            check_stack(lua.state, 2)?;
200
201            lua.push_ref(&self.0);
202            let thread_state = ffi::lua_tothread(lua.state, -1);
203
204            #[cfg(all(feature = "lua54", not(feature = "vendored")))]
205            let status = ffi::lua_resetthread(thread_state);
206            #[cfg(all(feature = "lua54", feature = "vendored"))]
207            let status = ffi::lua_closethread(thread_state, lua.state);
208            #[cfg(feature = "lua54")]
209            if status != ffi::LUA_OK {
210                return Err(pop_error(thread_state, status));
211            }
212            #[cfg(all(feature = "luajit", feature = "vendored"))]
213            ffi::lua_resetthread(lua.state, thread_state);
214            #[cfg(feature = "luau")]
215            ffi::lua_resetthread(thread_state);
216
217            lua.push_ref(&func.0);
218            ffi::lua_xmove(lua.state, thread_state, 1);
219
220            #[cfg(feature = "luau")]
221            {
222                // Inherit `LUA_GLOBALSINDEX` from the caller
223                ffi::lua_xpush(lua.state, thread_state, ffi::LUA_GLOBALSINDEX);
224                ffi::lua_replace(thread_state, ffi::LUA_GLOBALSINDEX);
225            }
226
227            Ok(())
228        }
229    }

231    /// Converts Thread to an AsyncThread which implements [`Future`] and [`Stream`] traits.
232    ///
233    /// `args` are passed as arguments to the thread function for first call.
234    /// The object calls [`resume()`] while polling and also allows to run rust futures
235    /// to completion using an executor.
236    ///
237    /// Using AsyncThread as a Stream allows to iterate through `coroutine.yield()`
238    /// values whereas Future version discards that values and poll until the final
239    /// one (returned from the thread function).
240    ///
241    /// Requires `feature = "async"`
242    ///
243    /// [`Future`]: futures_core::future::Future
244    /// [`Stream`]: futures_core::stream::Stream
245    /// [`resume()`]: https://www.lua.org/manual/5.4/manual.html#lua_resume
246    ///
247    /// # Examples
248    ///
249    /// ```
250    /// # use mlua::{Lua, Result, Thread};
251    /// use futures::stream::TryStreamExt;
252    /// # #[tokio::main]
253    /// # async fn main() -> Result<()> {
254    /// # let lua = Lua::new();
255    /// let thread: Thread = lua.load(r#"
256    ///     coroutine.create(function (sum)
257    ///         for i = 1,10 do
258    ///             sum = sum + i
259    ///             coroutine.yield(sum)
260    ///         end
261    ///         return sum
262    ///     end)
263    /// "#).eval()?;
264    ///
265    /// let mut stream = thread.into_async::<_, i64>(1);
266    /// let mut sum = 0;
267    /// while let Some(n) = stream.try_next().await? {
268    ///     sum += n;
269    /// }
270    ///
271    /// assert_eq!(sum, 286);
272    ///
273    /// # Ok(())
274    /// # }
275    /// ```
276    #[cfg(feature = "async")]
277    #[cfg_attr(docsrs, doc(cfg(feature = "async")))]
278    pub fn into_async<A, R>(self, args: A) -> AsyncThread<'lua, R>
279    where
280        A: ToLuaMulti<'lua>,
281        R: FromLuaMulti<'lua>,
282    {
283        let args = args.to_lua_multi(self.0.lua);
284        AsyncThread {
285            thread: self,
286            args0: RefCell::new(Some(args)),
287            ret: PhantomData,
288            recycle: false,
289        }
290    }

292    /// Enables sandbox mode on this thread.
293    ///
294    /// Under the hood replaces the global environment table with a new table,
295    /// that performs writes locally and proxies reads to caller's global environment.
296    ///
297    /// This mode ideally should be used together with the global sandbox mode [`Lua::sandbox()`].
298    ///
299    /// Please note that Luau links environment table with chunk when loading it into Lua state.
300    /// Therefore you need to load chunks into a thread to link with the thread environment.
301    ///
302    /// # Examples
303    ///
304    /// ```
305    /// # use mlua::{Lua, Result};
306    /// # fn main() -> Result<()> {
307    /// let lua = Lua::new();
308    /// let thread = lua.create_thread(lua.create_function(|lua2, ()| {
309    ///     lua2.load("var = 123").exec()?;
310    ///     assert_eq!(lua2.globals().get::<_, u32>("var")?, 123);
311    ///     Ok(())
312    /// })?)?;
313    /// thread.sandbox()?;
314    /// thread.resume(())?;
315    ///
316    /// // The global environment should be unchanged
317    /// assert_eq!(lua.globals().get::<_, Option<u32>>("var")?, None);
318    /// # Ok(())
319    /// # }
320    /// ```
321    ///
322    /// Requires `feature = "luau"`
323    #[cfg(any(feature = "luau", docsrs))]
324    #[cfg_attr(docsrs, doc(cfg(feature = "luau")))]
325    #[doc(hidden)]
326    pub fn sandbox(&self) -> Result<()> {
327        let lua = self.0.lua;
328        unsafe {
329            let thread = ffi::lua_tothread(lua.ref_thread(), self.0.index);
330            check_stack(thread, 1)?;
331            check_stack(lua.state, 3)?;
332            // Inherit `LUA_GLOBALSINDEX` from the caller
333            ffi::lua_xpush(lua.state, thread, ffi::LUA_GLOBALSINDEX);
334            ffi::lua_replace(thread, ffi::LUA_GLOBALSINDEX);
335            protect_lua!(lua.state, 0, 0, |_| ffi::luaL_sandboxthread(thread))
336        }
337    }
338}

354#[cfg(feature = "async")]
355#[cfg(any(
356    feature = "lua54",
357    all(feature = "luajit", feature = "vendored"),
358    feature = "luau",
359))]
360impl<'lua, R> Drop for AsyncThread<'lua, R> {
361    fn drop(&mut self) {
362        if self.recycle {
363            unsafe {
364                let lua = self.thread.0.lua;
365                // For Lua 5.4 this also closes all pending to-be-closed variables
366                if !lua.recycle_thread(&mut self.thread) {
367                    #[cfg(feature = "lua54")]
368                    if self.thread.status() == ThreadStatus::Error {
369                        let thread_state = ffi::lua_tothread(lua.ref_thread(), self.thread.0.index);
370                        ffi::lua_resetthread(thread_state);
371                    }
372                }
373            }
374        }
375    }
376}

378#[cfg(feature = "async")]
379impl<'lua, R> Stream for AsyncThread<'lua, R>
380where
381    R: FromLuaMulti<'lua>,
382{
383    type Item = Result<R>;
384
385    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
386        let lua = self.thread.0.lua;
387
388        match self.thread.status() {
389            ThreadStatus::Resumable => {}
390            _ => return Poll::Ready(None),
391        };
392
393        let _wg = WakerGuard::new(lua, cx.waker().clone());
394        let ret: MultiValue = if let Some(args) = self.args0.borrow_mut().take() {
395            self.thread.resume(args?)?
396        } else {
397            self.thread.resume(())?
398        };
399
400        if is_poll_pending(&ret) {
401            return Poll::Pending;
402        }
403
404        cx.waker().wake_by_ref();
405        Poll::Ready(Some(R::from_lua_multi(ret, lua)))
406    }
407}

409#[cfg(feature = "async")]
410impl<'lua, R> Future for AsyncThread<'lua, R>
411where
412    R: FromLuaMulti<'lua>,
413{
414    type Output = Result<R>;
415
416    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
417        let lua = self.thread.0.lua;
418
419        match self.thread.status() {
420            ThreadStatus::Resumable => {}
421            _ => return Poll::Ready(Err(Error::CoroutineInactive)),
422        };
423
424        let _wg = WakerGuard::new(lua, cx.waker().clone());
425        let ret: MultiValue = if let Some(args) = self.args0.borrow_mut().take() {
426            self.thread.resume(args?)?
427        } else {
428            self.thread.resume(())?
429        };
430
431        if is_poll_pending(&ret) {
432            return Poll::Pending;
433        }
434
435        if let ThreadStatus::Resumable = self.thread.status() {
436            // Ignore value returned via yield()
437            cx.waker().wake_by_ref();
438            return Poll::Pending;
439        }
440
441        Poll::Ready(R::from_lua_multi(ret, lua))
442    }
443}

445#[cfg(feature = "async")]
446#[inline(always)]
447fn is_poll_pending(val: &MultiValue) -> bool {
448    match val.iter().enumerate().last() {
449        Some((0, Value::LightUserData(ud))) => {
450            std::ptr::eq(ud.0 as *const u8, &ASYNC_POLL_PENDING as *const u8)
451        }
452        _ => false,
453    }
454}
