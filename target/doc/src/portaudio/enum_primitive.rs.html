26/// Helper macro for internal use by `enum_from_primitive!`.
27macro_rules! enum_from_primitive_impl_ty {
28    ($meth:ident, $ty:ty, $name:ident, $( $variant:ident )*) => {
29        #[allow(non_upper_case_globals, unused)]
30        fn $meth(n: $ty) -> $crate::Option<Self> {
31            $( if n == $name::$variant as $ty {
32                $crate::Option::Some($name::$variant)
33            } else )* {
34                $crate::Option::None
35            }
36        }
37    };
38}
39
40/// Helper macro for internal use by `enum_from_primitive!`.
41macro_rules! enum_from_primitive_impl {
42    ($name:ident, $( $variant:ident )*) => {
43        impl $crate::FromPrimitive for $name {
44            enum_from_primitive_impl_ty! { from_i64, i64, $name, $( $variant )* }
45            enum_from_primitive_impl_ty! { from_u64, u64, $name, $( $variant )* }
46        }
47    };
48}
49
50/// Wrap this macro around an `enum` declaration to get an
51/// automatically generated implementation of `num::FromPrimitive`.
52macro_rules! enum_from_primitive {
53    (
54        $( #[$enum_attr:meta] )*
55            pub enum $name:ident {
56                $( #[$variant_attr:meta]
57                     $variant:ident = $discriminator:expr ),*,
58            }
59    ) => {
60        $( #[$enum_attr] )*
61            pub enum $name {
62                $( #[$variant_attr]
63                     $variant = $discriminator ),*,
64            }
65        enum_from_primitive_impl! { $name, $( $variant )+ }
66    };
