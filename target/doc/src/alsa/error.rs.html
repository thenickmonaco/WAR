8/// ALSA error
9///
10/// Most ALSA functions can return a negative error code.
11/// If so, then that error code is wrapped into this `Error` struct.
12/// An Error is also returned in case ALSA returns a string that
13/// cannot be translated into Rust's UTF-8 strings.
14#[derive(Debug, Clone, PartialEq, Copy)]
15pub struct Error(&'static str, nix::Error);
16
17pub type Result<T> = ::std::result::Result<T, Error>;

19macro_rules! acheck {
20    ($f: ident ( $($x: expr),* ) ) => {{
21        let r = unsafe { alsa::$f( $($x),* ) };
22        if r < 0 { Err(Error::new(stringify!($f), -r as ::libc::c_int)) }
23        else { Ok(r) }
24    }}

27pub fn from_const<'a>(func: &'static str, s: *const c_char) -> Result<&'a str> {
28    if s.is_null() { return Err(invalid_str(func)) };
29    let cc = unsafe { CStr::from_ptr(s) };
30    str::from_utf8(cc.to_bytes()).map_err(|_| invalid_str(func))
31}
32
33pub fn from_alloc(func: &'static str, s: *mut c_char) -> Result<String> {
34    if s.is_null() { return Err(invalid_str(func)) };
35    let c = unsafe { CStr::from_ptr(s) };
36    let ss = str::from_utf8(c.to_bytes()).map_err(|_| {
37        unsafe { free(s as *mut c_void); }
38        invalid_str(func)
39    })?.to_string();
40    unsafe { free(s as *mut c_void); }
41    Ok(ss)
42}
43
44pub fn from_code(func: &'static str, r: c_int) -> Result<c_int> {
45    if r < 0 { Err(Error::new(func, r)) }
46    else { Ok(r) }
47}

49impl Error {

55    pub fn unsupported(func: &'static str) -> Error {
