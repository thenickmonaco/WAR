13/// A description of a given subglyph returned by `GlyphSlot::get_subglyph_info`
14/// function.
15#[derive(Copy, Clone)]
16pub struct SubGlyphInfo {
17    /// The glyph index of the subglyph.
18    pub index: i32,
19    /// The subglyph flags, see FT_SUBGLYPH_FLAG_XXX.
20    pub flags: u32,
21    /// The subglyph's first argument (if any).
22    pub arg1: i32,
23    /// The subglyph's second argument (if any).
24    pub arg2: i32,
25    /// The subglyph transformation (if any).
26    pub transfrom: ffi::FT_Matrix
27}
28
29impl Default for SubGlyphInfo {
30    fn default() -> Self {
31        SubGlyphInfo {
32            index: 0,
33            flags: 0,
34            arg1: 0,
35            arg2: 0,
36            transfrom: ffi::FT_Matrix { xx: 0, xy: 0, yx: 0, yy: 0 }
37        }
38    }
39}

41/// A struct encapsulating the space for a glyph within a `Library`
42#[derive(Copy, Clone, Eq, PartialEq, Hash)]
43pub struct GlyphSlot {
44    library_raw: ffi::FT_Library,
45    raw: ffi::FT_GlyphSlot
46}
47
48impl GlyphSlot {
49    /// Create a `GlyphSlot` from its constituent C parts
50    pub unsafe fn from_raw(library_raw: ffi::FT_Library, raw: ffi::FT_GlyphSlot) -> Self {
51        GlyphSlot {
52            library_raw: library_raw,
53            raw: raw
54        }
55    }
56
57    /// Convert a given glyph image to a bitmap. It does so by inspecting the glyph image format,
58    /// finding the relevant renderer, and invoking it.
59    pub fn render_glyph(&self, render_mode: RenderMode) -> FtResult<()> {
60        let err = unsafe {
61            ffi::FT_Render_Glyph(self.raw, render_mode as u32)
62        };
63        if err == ffi::FT_Err_Ok {
64            Ok(())
65        } else {
66            Err(err.into())
67        }
68    }
69
70    /// Retrieve a description of a given subglyph. Only use it if the glyph's format is
71    /// FT_GLYPH_FORMAT_COMPOSITE; an error is returned otherwise.
72    pub fn get_subglyph_info(&self, sub_index: u32) -> FtResult<SubGlyphInfo> {
73        let mut info = SubGlyphInfo::default();
74        let err = unsafe {
75            ffi::FT_Get_SubGlyph_Info(self.raw, sub_index, &mut info.index, &mut info.flags,
76                                      &mut info.arg1, &mut info.arg2, &mut info.transfrom)
77        };
78        if err == ffi::FT_Err_Ok {
79            Ok(info)
80        } else {
81            Err(err.into())
82        }
83    }
84
85    /// Returns a glyph object, that is similar to a `GlyphSlot` but managed outside of the library
86    pub fn get_glyph(&self) -> FtResult<Glyph> {
87        let mut aglyph = null_mut();
88
89        let err = unsafe {
90            ffi::FT_Get_Glyph(self.raw, &mut aglyph)
91        };
92        if err == ffi::FT_Err_Ok {
93            Ok(unsafe { Glyph::from_raw(self.library_raw, aglyph) })
94        } else {
95            Err(err.into())
96        }
97    }
98
99    /// In freetype, the `Outline` object is a scalable glyph. This method unpacks a glyph into
100    /// this object, or returns `None` if the glyph has no `outline`
101    pub fn outline(&self) -> Option<Outline> {
102        let outline = unsafe { &(*self.raw).outline };
103        let format = unsafe { (*self.raw).format };
104
105        if format == ffi::FT_GLYPH_FORMAT_OUTLINE {
106            let outline = unsafe {
107                Outline::from_raw(outline)
108            };
109            Some(outline)
110        } else {
111            None
112        }
113    }

116    /// This field is used as a bitmap descriptor when the slot format is FT_GLYPH_FORMAT_BITMAP.
117    /// Note that the address and content of the bitmap buffer can change between calls of
118    /// FT_Load_Glyph and a few other functions.
119    #[inline(always)]
120    pub fn bitmap(&self) -> Bitmap {
121        unsafe { Bitmap::from_raw(&(*self.raw).bitmap) }
122    }
123
124    /// The bitmap's left bearing expressed in integer pixels. Only valid if the format is
125    /// FT_GLYPH_FORMAT_BITMAP, this is, if the glyph slot contains a bitmap.
126    #[inline(always)]
127    pub fn bitmap_left(&self) -> i32 {
128        unsafe {
129            (*self.raw).bitmap_left
130        }
131    }
132
133    /// The bitmap's top bearing expressed in integer pixels. Remember that this is the distance
134    /// from the baseline to the top-most glyph scanline, upwards y coordinates being positive.
135    #[inline(always)]
136    pub fn bitmap_top(&self) -> i32 {
137        unsafe {
138            (*self.raw).bitmap_top
139        }
140    }
141
142    /// This shorthand is, depending on FT_LOAD_IGNORE_TRANSFORM, the transformed (hinted) advance
143    /// width for the glyph, in 26.6 fractional pixel format. As specified with
144    /// FT_LOAD_VERTICAL_LAYOUT, it uses either the ‘horiAdvance’ or the ‘vertAdvance’ value of
145    /// ‘metrics’ field.
146    #[inline(always)]
147    pub fn advance(&self) -> Vector {
148        unsafe {
149            (*self.raw).advance
150        }
151    }
152
153    /// The advance width of the unhinted glyph. Its value is expressed in 16.16 fractional pixels,
154    /// unless FT_LOAD_LINEAR_DESIGN is set when loading the glyph. This field can be important to
155    /// perform correct WYSIWYG layout. Only relevant for outline glyphs.
156    #[inline(always)]
157    pub fn linear_hori_advance(&self) -> ffi::FT_Fixed {
158        unsafe {
159            (*self.raw).linearHoriAdvance
160        }
161    }
162
163    /// The advance height of the unhinted glyph. Its value is expressed in 16.16 fractional
164    /// pixels, unless FT_LOAD_LINEAR_DESIGN is set when loading the glyph. This field can be
165    /// important to perform correct WYSIWYG layout. Only relevant for outline glyphs.
166    #[inline(always)]
167    pub fn linear_vert_advance(&self) -> ffi::FT_Fixed {
168        unsafe {
169            (*self.raw).linearVertAdvance
170        }
171    }
172
173    /// The metrics of the last loaded glyph in the slot. The returned values depend on the last
174    /// load flags (see the FT_Load_Glyph API function) and can be expressed either in 26.6
175    /// fractional pixels or font units.
176    #[inline(always)]
177    pub fn metrics(&self) -> GlyphMetrics {
178        unsafe {
179            (*self.raw).metrics
180        }
181    }
182
183    /// Get a pointer to the underlying c struct
184    #[inline(always)]
185    pub fn raw(&self) -> &ffi::FT_GlyphSlotRec {
186        unsafe {
187            &*self.raw
188        }
189    }
