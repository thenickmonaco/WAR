45    /// Calls the function, passing `args` as function arguments.
46    ///
47    /// The function's return values are converted to the generic type `R`.
48    ///
49    /// # Examples
50    ///
51    /// Call Lua's built-in `tostring` function:
52    ///
53    /// ```
54    /// # use mlua::{Function, Lua, Result};
55    /// # fn main() -> Result<()> {
56    /// # let lua = Lua::new();
57    /// let globals = lua.globals();
58    ///
59    /// let tostring: Function = globals.get("tostring")?;
60    ///
61    /// assert_eq!(tostring.call::<_, String>(123)?, "123");
62    ///
63    /// # Ok(())
64    /// # }
65    /// ```
66    ///
67    /// Call a function with multiple arguments:
68    ///
69    /// ```
70    /// # use mlua::{Function, Lua, Result};
71    /// # fn main() -> Result<()> {
72    /// # let lua = Lua::new();
73    /// let sum: Function = lua.load(
74    ///     r#"
75    ///         function(a, b)
76    ///             return a + b
77    ///         end
78    /// "#).eval()?;
79    ///
80    /// assert_eq!(sum.call::<_, u32>((3, 4))?, 3 + 4);
81    ///
82    /// # Ok(())
83    /// # }
84    /// ```
85    pub fn call<A: ToLuaMulti<'lua>, R: FromLuaMulti<'lua>>(&self, args: A) -> Result<R> {
86        let lua = self.0.lua;
87
88        let mut args = args.to_lua_multi(lua)?;
89        let nargs = args.len() as c_int;
90
91        let results = unsafe {
92            let _sg = StackGuard::new(lua.state);
93            check_stack(lua.state, nargs + 3)?;
94
95            ffi::lua_pushcfunction(lua.state, error_traceback);
96            let stack_start = ffi::lua_gettop(lua.state);
97            lua.push_ref(&self.0);
98            for arg in args.drain_all() {
99                lua.push_value(arg)?;
100            }
101            let ret = ffi::lua_pcall(lua.state, nargs, ffi::LUA_MULTRET, stack_start);
102            if ret != ffi::LUA_OK {
103                return Err(pop_error(lua.state, ret));
104            }
105            let nresults = ffi::lua_gettop(lua.state) - stack_start;
106            let mut results = args; // Reuse MultiValue container
107            assert_stack(lua.state, 2);
108            for _ in 0..nresults {
109                results.push_front(lua.pop_value());
110            }
111            ffi::lua_pop(lua.state, 1);
112            results
113        };
114        R::from_lua_multi(results, lua)
115    }

117    /// Returns a future that, when polled, calls `self`, passing `args` as function arguments,
118    /// and drives the execution.
119    ///
120    /// Internally it wraps the function to an [`AsyncThread`].
121    ///
122    /// Requires `feature = "async"`
123    ///
124    /// # Examples
125    ///
126    /// ```
127    /// use std::time::Duration;
128    /// use futures_timer::Delay;
129    /// # use mlua::{Lua, Result};
130    /// # #[tokio::main]
131    /// # async fn main() -> Result<()> {
132    /// # let lua = Lua::new();
133    ///
134    /// let sleep = lua.create_async_function(move |_lua, n: u64| async move {
135    ///     Delay::new(Duration::from_millis(n)).await;
136    ///     Ok(())
137    /// })?;
138    ///
139    /// sleep.call_async(10).await?;
140    ///
141    /// # Ok(())
142    /// # }
143    /// ```
144    ///
145    /// [`AsyncThread`]: crate::AsyncThread
146    #[cfg(feature = "async")]
147    #[cfg_attr(docsrs, doc(cfg(feature = "async")))]
148    pub fn call_async<'fut, A, R>(&self, args: A) -> LocalBoxFuture<'fut, Result<R>>
149    where
150        'lua: 'fut,
151        A: ToLuaMulti<'lua>,
152        R: FromLuaMulti<'lua> + 'fut,
153    {
154        let lua = self.0.lua;
155        match lua.create_recycled_thread(self.clone()) {
156            Ok(t) => {
157                let mut t = t.into_async(args);
158                t.set_recyclable(true);
159                Box::pin(t)
160            }
161            Err(e) => Box::pin(future::err(e)),
162        }
163    }

165    /// Returns a function that, when called, calls `self`, passing `args` as the first set of
166    /// arguments.
167    ///
168    /// If any arguments are passed to the returned function, they will be passed after `args`.
169    ///
170    /// # Examples
171    ///
172    /// ```
173    /// # use mlua::{Function, Lua, Result};
174    /// # fn main() -> Result<()> {
175    /// # let lua = Lua::new();
176    /// let sum: Function = lua.load(
177    ///     r#"
178    ///         function(a, b)
179    ///             return a + b
180    ///         end
181    /// "#).eval()?;
182    ///
183    /// let bound_a = sum.bind(1)?;
184    /// assert_eq!(bound_a.call::<_, u32>(2)?, 1 + 2);
185    ///
186    /// let bound_a_and_b = sum.bind(13)?.bind(57)?;
187    /// assert_eq!(bound_a_and_b.call::<_, u32>(())?, 13 + 57);
188    ///
189    /// # Ok(())
190    /// # }
191    /// ```
192    pub fn bind<A: ToLuaMulti<'lua>>(&self, args: A) -> Result<Function<'lua>> {
193        unsafe extern "C" fn args_wrapper_impl(state: *mut ffi::lua_State) -> c_int {
194            let nargs = ffi::lua_gettop(state);
195            let nbinds = ffi::lua_tointeger(state, ffi::lua_upvalueindex(1)) as c_int;
196            ffi::luaL_checkstack(state, nbinds, ptr::null());
197
198            for i in 0..nbinds {
199                ffi::lua_pushvalue(state, ffi::lua_upvalueindex(i + 2));
200            }
201            if nargs > 0 {
202                ffi::lua_rotate(state, 1, nbinds);
203            }
204
205            nargs + nbinds
206        }
207
208        let lua = self.0.lua;
209
210        let args = args.to_lua_multi(lua)?;
211        let nargs = args.len() as c_int;
212
213        if nargs == 0 {
214            return Ok(self.clone());
215        }
216
217        if nargs + 1 > ffi::LUA_MAX_UPVALUES {
218            return Err(Error::BindError);
219        }
220
221        let args_wrapper = unsafe {
222            let _sg = StackGuard::new(lua.state);
223            check_stack(lua.state, nargs + 3)?;
224
225            ffi::lua_pushinteger(lua.state, nargs as ffi::lua_Integer);
226            for arg in args {
227                lua.push_value(arg)?;
228            }
229            protect_lua!(lua.state, nargs + 1, 1, fn(state) {
230                ffi::lua_pushcclosure(state, args_wrapper_impl, ffi::lua_gettop(state));
231            })?;
232
233            Function(lua.pop_ref())
234        };
235
236        lua.load(
237            r#"
238            local func, args_wrapper = ...
239            return function(...)
240                return func(args_wrapper(...))
241            end
242            "#,
243        )
244        .try_cache()
245        .set_name("_mlua_bind")?
246        .call((self.clone(), args_wrapper))
247    }

249    /// Returns information about the function.
250    ///
251    /// Corresponds to the `>Sn` what mask for [`lua_getinfo`] when applied to the function.
252    ///
253    /// [`lua_getinfo`]: https://www.lua.org/manual/5.4/manual.html#lua_getinfo
254    pub fn info(&self) -> FunctionInfo {
255        let lua = self.0.lua;
256        unsafe {
257            let _sg = StackGuard::new(lua.state);
258            assert_stack(lua.state, 1);
259
260            let mut ar: ffi::lua_Debug = mem::zeroed();
261            lua.push_ref(&self.0);
262            #[cfg(not(feature = "luau"))]
263            let res = ffi::lua_getinfo(lua.state, cstr!(">Sn"), &mut ar);
264            #[cfg(feature = "luau")]
265            let res = ffi::lua_getinfo(lua.state, -1, cstr!("sn"), &mut ar);
266            mlua_assert!(res != 0, "lua_getinfo failed with `>Sn`");
267
268            FunctionInfo {
269                name: ptr_to_cstr_bytes(ar.name).map(|s| s.to_vec()),
270                #[cfg(not(feature = "luau"))]
271                name_what: ptr_to_cstr_bytes(ar.namewhat).map(|s| s.to_vec()),
272                #[cfg(feature = "luau")]
273                name_what: None,
274                what: ptr_to_cstr_bytes(ar.what).map(|s| s.to_vec()),
275                source: ptr_to_cstr_bytes(ar.source).map(|s| s.to_vec()),
276                #[cfg(not(feature = "luau"))]
277                short_src: ptr_to_cstr_bytes(ar.short_src.as_ptr()).map(|s| s.to_vec()),
278                #[cfg(feature = "luau")]
279                short_src: ptr_to_cstr_bytes(ar.short_src).map(|s| s.to_vec()),
280                line_defined: ar.linedefined,
281                #[cfg(not(feature = "luau"))]
282                last_line_defined: ar.lastlinedefined,
283            }
284        }
285    }

287    /// Dumps the function as a binary chunk.
288    ///
289    /// If `strip` is true, the binary representation may not include all debug information
290    /// about the function, to save space.
291    ///
292    /// For Luau a [Compiler] can be used to compile Lua chunks to bytecode.
293    ///
294    /// [Compiler]: crate::chunk::Compiler
295    #[cfg(not(feature = "luau"))]
296    #[cfg_attr(docsrs, doc(cfg(not(feature = "luau"))))]
297    pub fn dump(&self, strip: bool) -> Vec<u8> {
298        unsafe extern "C" fn writer(
299            _state: *mut ffi::lua_State,
300            buf: *const c_void,
301            buf_len: usize,
302            data: *mut c_void,
303        ) -> c_int {
304            let data = &mut *(data as *mut Vec<u8>);
305            let buf = slice::from_raw_parts(buf as *const u8, buf_len);
306            data.extend_from_slice(buf);
307            0
308        }
309
310        let lua = self.0.lua;
311        let mut data: Vec<u8> = Vec::new();
312        unsafe {
313            let _sg = StackGuard::new(lua.state);
314            assert_stack(lua.state, 1);
315
316            lua.push_ref(&self.0);
317            let data_ptr = &mut data as *mut Vec<u8> as *mut c_void;
318            ffi::lua_dump(lua.state, writer, data_ptr, strip as i32);
319            ffi::lua_pop(lua.state, 1);
320        }
321
322        data
323    }

325    /// Retrieves recorded coverage information about this Lua function including inner calls.
326    ///
327    /// This function takes a callback as an argument and calls it providing [`CoverageInfo`] snapshot
328    /// per each executed inner function.
329    ///
330    /// Recording of coverage information is controlled by [`Compiler::set_coverage_level`] option.
331    ///
332    /// Requires `feature = "luau"`
333    ///
334    /// [`Compiler::set_coverage_level`]: crate::chunk::Compiler::set_coverage_level
335    #[cfg(any(feature = "luau", docsrs))]
336    #[cfg_attr(docsrs, doc(cfg(feature = "luau")))]
337    pub fn coverage<F>(&self, mut func: F)
338    where
339        F: FnMut(CoverageInfo),
340    {
341        use std::ffi::CStr;
342        use std::os::raw::c_char;
343
344        unsafe extern "C" fn callback<F: FnMut(CoverageInfo)>(
345            data: *mut c_void,
346            function: *const c_char,
347            line_defined: c_int,
348            depth: c_int,
349            hits: *const c_int,
350            size: usize,
351        ) {
352            let function = if !function.is_null() {
353                Some(CStr::from_ptr(function).to_string_lossy().to_string())
354            } else {
355                None
356            };
357            let rust_callback = &mut *(data as *mut F);
358            rust_callback(CoverageInfo {
359                function,
360                line_defined,
361                depth,
362                hits: slice::from_raw_parts(hits, size).to_vec(),
363            });
364        }
365
366        let lua = self.0.lua;
367        unsafe {
368            let _sg = StackGuard::new(lua.state);
369            assert_stack(lua.state, 1);
370
371            lua.push_ref(&self.0);
372            let func_ptr = &mut func as *mut F as *mut c_void;
373            ffi::lua_getcoverage(lua.state, -1, func_ptr, callback::<F>);
374        }
375    }
