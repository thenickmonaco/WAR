23/// A dynamically typed Lua value. The `String`, `Table`, `Function`, `Thread`, and `UserData`
24/// variants contain handle types into the internal Lua state. It is a logic error to mix handle
25/// types between separate `Lua` instances, and doing so will result in a panic.
26#[derive(Debug, Clone)]
27pub enum Value<'lua> {
28    /// The Lua value `nil`.
29    Nil,
30    /// The Lua value `true` or `false`.
31    Boolean(bool),
32    /// A "light userdata" object, equivalent to a raw pointer.
33    LightUserData(LightUserData),
34    /// An integer number.
35    ///
36    /// Any Lua number convertible to a `Integer` will be represented as this variant.
37    Integer(Integer),
38    /// A floating point number.
39    Number(Number),
40    /// A Luau vector.
41    #[cfg(any(feature = "luau", doc))]
42    #[cfg_attr(docsrs, doc(cfg(feature = "luau")))]
43    Vector(f32, f32, f32),
44    /// An interned string, managed by Lua.
45    ///
46    /// Unlike Rust strings, Lua strings may not be valid UTF-8.
47    String(String<'lua>),
48    /// Reference to a Lua table.
49    Table(Table<'lua>),
50    /// Reference to a Lua function (or closure).
51    Function(Function<'lua>),
52    /// Reference to a Lua thread (or coroutine).
53    Thread(Thread<'lua>),
54    /// Reference to a userdata object that holds a custom type which implements `UserData`.
55    /// Special builtin userdata types will be represented as other `Value` variants.
56    UserData(AnyUserData<'lua>),
57    /// `Error` is a special builtin userdata type. When received from Lua it is implicitly cloned.
58    Error(Error),
59}

63impl<'lua> Value<'lua> {
64    pub const fn type_name(&self) -> &'static str {
65        match *self {
66            Value::Nil => "nil",
67            Value::Boolean(_) => "boolean",
68            Value::LightUserData(_) => "lightuserdata",
69            Value::Integer(_) => "integer",
70            Value::Number(_) => "number",
71            #[cfg(feature = "luau")]
72            Value::Vector(_, _, _) => "vector",
73            Value::String(_) => "string",
74            Value::Table(_) => "table",
75            Value::Function(_) => "function",
76            Value::Thread(_) => "thread",
77            Value::UserData(_) => "userdata",
78            Value::Error(_) => "error",
79        }
80    }
81
82    /// Compares two values for equality.
83    ///
84    /// Equality comparisons do not convert strings to numbers or vice versa.
85    /// Tables, Functions, Threads, and Userdata are compared by reference:
86    /// two objects are considered equal only if they are the same object.
87    ///
88    /// If Tables or Userdata have `__eq` metamethod then mlua will try to invoke it.
89    /// The first value is checked first. If that value does not define a metamethod
90    /// for `__eq`, then mlua will check the second value.
91    /// Then mlua calls the metamethod with the two values as arguments, if found.
92    pub fn equals<T: AsRef<Self>>(&self, other: T) -> Result<bool> {
93        match (self, other.as_ref()) {
94            (Value::Table(a), Value::Table(b)) => a.equals(b),
95            (Value::UserData(a), Value::UserData(b)) => a.equals(b),
96            _ => Ok(self == other.as_ref()),
97        }
98    }
99
100    /// Converts the value to a generic C pointer.
101    ///
102    /// The value can be a userdata, a table, a thread, a string, or a function; otherwise it returns NULL.
103    /// Different objects will give different pointers.
104    /// There is no way to convert the pointer back to its original value.
105    ///
106    /// Typically this function is used only for hashing and debug information.
107    #[inline]
108    pub fn to_pointer(&self) -> *const c_void {
109        unsafe {
110            match self {
111                Value::LightUserData(ud) => ud.0,
112                Value::Table(t) => t.to_pointer(),
113                Value::String(s) => s.to_pointer(),
114                Value::Function(Function(r))
115                | Value::Thread(Thread(r))
116                | Value::UserData(AnyUserData(r)) => {
117                    ffi::lua_topointer(r.lua.ref_thread(), r.index)
118                }
119                _ => ptr::null(),
120            }
121        }
122    }
123}

125impl<'lua> PartialEq for Value<'lua> {
126    fn eq(&self, other: &Self) -> bool {
127        match (self, other) {
128            (Value::Nil, Value::Nil) => true,
129            (Value::Boolean(a), Value::Boolean(b)) => a == b,
130            (Value::LightUserData(a), Value::LightUserData(b)) => a == b,
131            (Value::Integer(a), Value::Integer(b)) => *a == *b,
132            (Value::Integer(a), Value::Number(b)) => *a as Number == *b,
133            (Value::Number(a), Value::Integer(b)) => *a == *b as Number,
134            (Value::Number(a), Value::Number(b)) => *a == *b,
135            #[cfg(feature = "luau")]
136            (Value::Vector(x1, y1, z1), Value::Vector(x2, y2, z2)) => (x1, y1, z1) == (x2, y2, z2),
137            (Value::String(a), Value::String(b)) => a == b,
138            (Value::Table(a), Value::Table(b)) => a == b,
139            (Value::Function(a), Value::Function(b)) => a == b,
140            (Value::Thread(a), Value::Thread(b)) => a == b,
141            (Value::UserData(a), Value::UserData(b)) => a == b,
142            _ => false,
143        }
144    }
145}

154#[cfg(feature = "serialize")]
155impl<'lua> Serialize for Value<'lua> {
156    fn serialize<S>(&self, serializer: S) -> StdResult<S::Ok, S::Error>
157    where
158        S: Serializer,
159    {
160        match self {
161            Value::Nil => serializer.serialize_unit(),
162            Value::Boolean(b) => serializer.serialize_bool(*b),
163            #[allow(clippy::useless_conversion)]
164            Value::Integer(i) => serializer
165                .serialize_i64((*i).try_into().expect("cannot convert lua_Integer to i64")),
166            #[allow(clippy::useless_conversion)]
167            Value::Number(n) => serializer.serialize_f64(*n),
168            #[cfg(feature = "luau")]
169            Value::Vector(x, y, z) => (x, y, z).serialize(serializer),
170            Value::String(s) => s.serialize(serializer),
171            Value::Table(t) => t.serialize(serializer),
172            Value::UserData(ud) => ud.serialize(serializer),
173            Value::LightUserData(ud) if ud.0.is_null() => serializer.serialize_none(),
174            Value::Error(_) | Value::LightUserData(_) | Value::Function(_) | Value::Thread(_) => {
175                let msg = format!("cannot serialize <{}>", self.type_name());
176                Err(ser::Error::custom(msg))
177            }
178        }
179    }
180}

347/// Trait for types that can be created from an arbitrary number of Lua values.
348///
349/// This is a generalization of `FromLua`, allowing an arbitrary number of Lua values to participate
350/// in the conversion. Any type that implements `FromLua` will automatically implement this trait.
351pub trait FromLuaMulti<'lua>: Sized {
352    /// Performs the conversion.
353    ///
354    /// In case `values` contains more values than needed to perform the conversion, the excess
355    /// values should be ignored. This reflects the semantics of Lua when calling a function or
356    /// assigning values. Similarly, if not enough values are given, conversions should assume that
357    /// any missing values are nil.
358    fn from_lua_multi(values: MultiValue<'lua>, lua: &'lua Lua) -> Result<Self>;
359}
