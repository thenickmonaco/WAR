4/// An enumeration type used to describe the format of pixels in a given bitmap. Note that
5/// additional formats may be added in the future.
6#[derive(Copy, Clone, Debug, PartialEq, Eq)]
7pub enum PixelMode {
8    /// This value is reserved.
9    None,
10
11    /// A monochrome bitmap, using 1 bit per pixel. Note that pixels are
12    /// stored in most-significant order (MSB), which means that the left-most
13    /// pixel in a byte has value 128.
14    Mono,
15
16    /// An 8-bit bitmap, generally used to represent anti-aliased glyph images.
17    /// Each pixel is stored in one byte. Note that the number of `gray`
18    /// levels is stored in the `num_grays` field of the FT_Bitmap structure
19    /// (it generally is 256).
20    Gray,
21
22    /// A 2-bit per pixel bitmap, used to represent embedded anti-aliased
23    /// bitmaps in font files according to the OpenType specification. We
24    /// haven't found a single font using this format, however.
25    Gray2,
26
27    /// A 4-bit per pixel bitmap, representing embedded anti-aliased bitmaps in
28    /// font files according to the OpenType specification. We haven't found a
29    /// single font using this format, however.
30    Gray4,
31
32    /// An 8-bit bitmap, representing RGB or BGR decimated glyph images used
33    /// for display on LCD displays; the bitmap is three times wider than the
34    /// original glyph image. See also FT_RENDER_MODE_LCD.
35    Lcd,
36
37    /// An 8-bit bitmap, representing RGB or BGR decimated glyph images used for
38    /// display on rotated LCD displays; the bitmap is three times taller than
39    /// the original glyph image. See also FT_RENDER_MODE_LCD_V.
40    LcdV,
41
42    /// An image with four 8-bit channels per pixel, representing a color image
43    /// (such as emoticons) with alpha channel. For each pixel, the format is
44    /// BGRA, which means, the blue channel comes first in memory. The color
45    /// channels are pre-multiplied and in the sRGB colorspace. For example,
46    /// full red at half-translucent opacity will be represented as
47    /// `00,00,80,80`, not `00,00,FF,80`. See also FT_LOAD_COLOR.
48    Bgra
49}

51#[allow(missing_copy_implementations)]
52pub struct Bitmap {
53    raw: *const ffi::FT_Bitmap
54}
55
56impl Bitmap {
57    pub unsafe fn from_raw(raw: *const ffi::FT_Bitmap) -> Self {
58        Bitmap {
59            raw: raw
60        }
61    }
62
63    /// A typeless pointer to the bitmap buffer. This value should be aligned
64    /// on 32-bit boundaries in most cases.
65    pub fn buffer(&self) -> &[u8] {
66        unsafe {
67            slice::from_raw_parts(
68                (*self.raw).buffer,
69                (self.pitch().abs() * self.rows()) as usize
70            )
71        }
72    }
73
74    /// The number of pixels in bitmap row.
75    pub fn width(&self) -> i32 {
76        unsafe {
77            (*self.raw).width
78        }
79    }
80
81    /// The number of bitmap rows.
82    pub fn rows(&self) -> i32 {
83        unsafe {
84            (*self.raw).rows
85        }
86    }
87
88    pub fn raw(&self) -> &ffi::FT_Bitmap {
89        unsafe {
90            &*self.raw
91        }
92    }
93
94    /// The pixel mode, i.e., how pixel bits are stored. See `PixelMode` for
95    /// possible values.
96    pub fn pixel_mode(&self) -> FtResult<PixelMode> {
97        let pixel_mode = unsafe { (*self.raw).pixel_mode } as u32;
98
99        Ok(match pixel_mode {
100            ffi::FT_PIXEL_MODE_NONE  => PixelMode::None,
101            ffi::FT_PIXEL_MODE_MONO  => PixelMode::Mono,
102            ffi::FT_PIXEL_MODE_GRAY  => PixelMode::Gray,
103            ffi::FT_PIXEL_MODE_GRAY2 => PixelMode::Gray2,
104            ffi::FT_PIXEL_MODE_GRAY4 => PixelMode::Gray4,
105            ffi::FT_PIXEL_MODE_LCD   => PixelMode::Lcd,
106            ffi::FT_PIXEL_MODE_LCD_V => PixelMode::LcdV,
107            ffi::FT_PIXEL_MODE_BGRA  => PixelMode::Bgra,
108            _ => return Err(Error::UnexpectedPixelMode)
109        })
110    }
111
112    /// The pitch's absolute value is the number of bytes taken by one bitmap row, including
113    /// padding. However, the pitch is positive when the bitmap has a ‘down’ flow, and negative
114    /// when it has an ‘up’ flow. In all cases, the pitch is an offset to add to a bitmap pointer
115    /// in order to go down one row.
116    ///
117    /// Note that ‘padding’ means the alignment of a bitmap to a byte border, and FreeType
118    /// functions normally align to the smallest possible integer value.
119    /// For the B/W rasterizer, ‘pitch’ is always an even number.
120    ///
121    /// To change the pitch of a bitmap (say, to make it a multiple of 4), use FT_Bitmap_Convert.
122    /// Alternatively, you might use callback functions to directly render to the application's
123    /// surface; see the file ‘example2.cpp’ in the tutorial for a demonstration.
124    pub fn pitch(&self) -> i32 {
125        unsafe {
126            (*self.raw).pitch
127        }
128    }
129}
