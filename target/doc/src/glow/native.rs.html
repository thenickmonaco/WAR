19    pub unsafe fn from_loader_function<F>(mut loader_function: F) -> Self
20    where
21        F: FnMut(&str) -> *const std::os::raw::c_void,
22    {
23        // Note(Lokathor): This is wildly inefficient, because the loader_function
24        // is doubtlessly just going to allocate the `&str` we pass into a new `CString`
25        // so that it can pass that `*const c_char` off to the OS's actual loader.
26        // However, this is the best we can do without changing the outer function
27        // signature into something that's less alloc crazy.
28        let raw: native_gl::GlFns =
29            native_gl::GlFns::load_with(|p: *const std::os::raw::c_char| {
30                let c_str = std::ffi::CStr::from_ptr(p);
31                loader_function(c_str.to_str().unwrap()) as *mut std::os::raw::c_void
32            });
33
34        // Retrieve and parse `GL_VERSION`
35        let raw_string = raw.GetString(VERSION);
36        let raw_version = std::ffi::CStr::from_ptr(raw_string as *const native_gl::GLchar)
37            .to_str()
38            .unwrap()
39            .to_owned();
40        let version = Version::parse(&raw_version).unwrap();
41
42        // Setup extensions and constants after the context has been built
43        let mut context = Self {
44            raw,
45            extensions: HashSet::new(),
46            constants: Constants::default(),
47            version,
48        };
49
50        // Use core-only functions to populate extension list
51        if (context.version >= Version::new(3, 0, None, String::from("")))
52            || (context.version >= Version::new_embedded(3, 0, String::from("")))
53        {
54            let num_extensions = context.get_parameter_i32(NUM_EXTENSIONS);
55            for i in 0..num_extensions {
56                let extension_name = context.get_parameter_indexed_string(EXTENSIONS, i as u32);
57                context.extensions.insert(extension_name);
58            }
59        } else {
60            // Fallback
61            context.extensions.extend(
62                context
63                    .get_parameter_string(EXTENSIONS)
64                    .split(' ')
65                    .map(|s| s.to_string()),
66            );
67        };
68
69        // After the extensions are known, we can populate constants (including
70        // constants that depend on extensions being enabled)
71        context.constants.max_label_length = if context.supports_debug() {
72            context.get_parameter_i32(MAX_LABEL_LENGTH)
73        } else {
74            0
75        };
76
77        context
78    }

360        let gl = &self.raw;

370        let mut length = 0;

376            uniform_max_size,

379            &mut utype,
380            name.as_ptr() as *mut native_gl::GLchar,
381        );
382        name.truncate(length as usize);
383
384        Some(ActiveUniform { size, utype, name })

2700        // Probe for the length of the name of the uniform block, and, failing

2702        // should be good enough for pretty much all contexts, including faulty
2703        // or partially faulty ones.

2716        let mut length = 0;
2717        gl.GetActiveUniformBlockName(
2718            program.0.get(),

2733                std::mem::align_of::<u8>(),
2734                std::mem::align_of::<native_gl::GLchar>(),
2735                "This operation is only safe in systems in which the alignment \
2736                of a GLchar is the same as that of an u8"
2737            );
2738            let buffer = std::slice::from_raw_parts(
2739                buffer.as_ptr() as *const u8,
2740                (length as usize + 1).min(buffer.len()),
2741            );
2742
2743            let name = CStr::from_bytes_with_nul(&buffer[..])
2744                .unwrap()
2745                .to_str()
2746                .unwrap()
2747                .to_owned();
2748
2749            name
2750        } else {
2751            String::from("")
2752        }
2753    }
2754}

2757    source: u32,
