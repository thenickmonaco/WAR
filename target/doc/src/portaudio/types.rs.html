32/// The type used to refer to audio devices.
33///
34/// Values of this type usually range from 0 to (PortAudio::device_count-1).
35#[derive(Copy, Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
36pub struct DeviceIndex(pub u32);
37
38/// The device to be used by some stream.
39///
40/// This is used as a field within the Settings for a **Stream**.
41#[derive(Copy, Clone, Debug, PartialEq, Eq)]
42pub enum DeviceKind {
43    /// An index to some Device.
44    Index(DeviceIndex),
45    /// Indicates that the device(s) to be used are specified in the host api specific stream info
46    /// structure.
47    UseHostApiSpecificDeviceSpecification,
48}
49
50impl From<DeviceIndex> for ffi::PaDeviceIndex {
51    fn from(idx: DeviceIndex) -> ffi::PaDeviceIndex {
52        let DeviceIndex(idx) = idx;
53        idx as ffi::PaDeviceIndex
54    }
55}
56
57impl From<DeviceIndex> for DeviceKind {
58    fn from(idx: DeviceIndex) -> DeviceKind {
59        DeviceKind::Index(idx)
60    }
61}
62
63impl From<DeviceKind> for ffi::PaDeviceIndex {
64    fn from(kind: DeviceKind) -> ffi::PaDeviceIndex {
65        match kind {
66            DeviceKind::Index(idx) => idx.into(),
67            DeviceKind::UseHostApiSpecificDeviceSpecification => -2,
68        }
69    }
70}

87/// A type used to dynamically represent the various standard sample formats (usually) supported by
88/// all PortAudio implementations.
89#[derive(Copy, Clone, Debug, PartialEq, Eq)]
90pub enum SampleFormat {
91    /// Uses -1.0 and +1.0 as the minimum and maximum respectively.
92    F32,
93    /// 32-bit signed integer sample representation.
94    I32,
95    /// 24-bit signed integer sample representation.
96    ///
97    /// TODO: Should work out how to support this properly.
98    I24,
99    /// 16-bit signed integer sample representation.
100    I16,
101    /// 8-bit signed integer sample representation.
102    I8,
103    /// An unsigned 8 bit format where 128 is considered "ground"
104    U8,
105    /// Some custom sample format.
106    ///
107    /// TODO: Need to work out how to support this properly. I was unable to find any official
108    /// info.
109    ///
110    /// The following e-mail by Bencina (2004) touches on the topic of custom formats:
111    ///
112    /// > "It is theoretically possible to pass "custom" data formats to PortAudio using the
113    /// paCustomFormat SampleFormat flag. I think the general idea is that when this bit is set,
114    /// the low word of the sample format byte is device specific. I know of no implementation that
115    /// has ever used this feature so it has not been extensively developed. That said, much of
116    /// PortAudio (V19 at least) assumes a frame based sample format, accomodating a block based
117    /// format such as mpeg would probably require bypassing some of the internal infrastructure
118    /// (such as the block adapter in pa_process). PortAudio has been designed for linear, frame
119    /// based i/o, so it would be up to you to propose/suggest ways in which to accomodate your
120    /// requirements." - http://music.columbia.edu/pipermail/portaudio/2004-February/003237.html
121    Custom,
122    /// This variant is used when none of the above variants can be inferred from a given
123    /// set of **SampleFormatFlags** via the `SampleFormat::from_flags` function.
124    Unknown,
125}
126
127impl SampleFormat {
128    /// Inspects the given **SampleFormatFlags** for the format.
129    ///
130    /// Returns `Some(SampleFormat)` if a matching format is found.
131    ///
132    /// Returns `None` if no matching format is found.
133    pub fn from_flags(flags: SampleFormatFlags) -> Self {
134        if flags.contains(SampleFormatFlags::FLOAT_32) {
135            SampleFormat::F32
136        } else if flags.contains(SampleFormatFlags::INT_32) {
137            SampleFormat::I32
138        } else if flags.contains(SampleFormatFlags::INT_24) {
139            SampleFormat::I24
140        } else if flags.contains(SampleFormatFlags::INT_16) {
141            SampleFormat::I16
142        } else if flags.contains(SampleFormatFlags::INT_8) {
143            SampleFormat::I8
144        } else if flags.contains(SampleFormatFlags::UINT_8) {
145            SampleFormat::U8
146        } else if flags.contains(SampleFormatFlags::CUSTOM_FORMAT) {
147            SampleFormat::Custom
148        } else {
149            SampleFormat::Unknown
150        }
151    }
152
153    /// Converts `self` into the respective **SampleFormatFlags**.
154    pub fn flags(self) -> SampleFormatFlags {
155        match self {
156            SampleFormat::F32 => SampleFormatFlags::FLOAT_32,
157            SampleFormat::I32 => SampleFormatFlags::INT_32,
158            SampleFormat::I24 => SampleFormatFlags::INT_24,
159            SampleFormat::I16 => SampleFormatFlags::INT_16,
160            SampleFormat::I8 => SampleFormatFlags::INT_8,
161            SampleFormat::U8 => SampleFormatFlags::UINT_8,
162            SampleFormat::Custom => SampleFormatFlags::CUSTOM_FORMAT,
163            SampleFormat::Unknown => SampleFormatFlags::empty(),
164        }
165    }
166
167    /// Returns the size of the **SampleFormat** in bytes.
168    ///
169    /// Returns `0` if the **SampleFormat** is **Custom** or **Unknown**.
170    pub fn size_in_bytes(&self) -> u8 {
171        match *self {
172            SampleFormat::F32 | SampleFormat::I32 => 4,
173            SampleFormat::I24 => 3,
174            SampleFormat::I16 => 2,
175            SampleFormat::I8 | SampleFormat::U8 => 1,
176            SampleFormat::Custom | SampleFormat::Unknown => 0,
177        }
178    }
179}

181pub mod sample_format_flags {
182    //! A type safe wrapper around PortAudio's `PaSampleFormat` flags.
183    use ffi;
184    bitflags! {
185        /// A type used to specify one or more sample formats. Each value indicates a possible
186        /// format for sound data passed to and from the stream callback, Pa_ReadStream and
187        /// Pa_WriteStream.
188        ///
189        /// The standard formats paFloat32, paInt16, paInt32, paInt24, paInt8 and aUInt8 are
190        /// usually implemented by all implementations.
191        ///
192        /// The floating point representation (FLOAT_32) uses +1.0 and -1.0 as the maximum and
193        /// minimum respectively.
194        ///
195        /// UINT_8 is an unsigned 8 bit format where 128 is considered "ground"
196        ///
197        /// The paNonInterleaved flag indicates that audio data is passed as an array of pointers
198        /// to separate buffers, one buffer for each channel. Usually, when this flag is not used,
199        /// audio data is passed as a single buffer with all channels interleaved.
200        pub struct SampleFormatFlags: ::std::os::raw::c_ulong {
201            /// 32 bits float sample format
202            const FLOAT_32 = ffi::PA_FLOAT_32;
203            /// 32 bits int sample format
204            const INT_32 = ffi::PA_INT_32;
205            /// Packed 24 bits int sample format
206            const INT_24 = ffi::PA_INT_24;
207            /// 16 bits int sample format
208            const INT_16 = ffi::PA_INT_16;
209            /// 8 bits int sample format
210            const INT_8 = ffi::PA_INT_8;
211            /// 8 bits unsigned int sample format
212            const UINT_8 = ffi::PA_UINT_8;
213            /// Custom sample format
214            const CUSTOM_FORMAT = ffi::PA_CUSTOM_FORMAT;
215            /// Non interleaved sample format
216            const NON_INTERLEAVED = ffi::PA_NON_INTERLEAVED;
217        }
218    }
219
220    impl From<ffi::SampleFormat> for SampleFormatFlags {
221        fn from(format: ffi::SampleFormat) -> Self {
222            SampleFormatFlags::from_bits(format).unwrap_or_else(|| SampleFormatFlags::empty())
223        }
224    }
225
226    impl ::std::fmt::Display for SampleFormatFlags {
227        fn fmt(&self, f: &mut ::std::fmt::Formatter) -> ::std::fmt::Result {
228            write!(
229                f,
230                "{:?}",
231                match self.bits() {
232                    ffi::PA_FLOAT_32 => "FLOAT_32",
233                    ffi::PA_INT_32 => "INT_32",
234                    //ffi::PA_INT_24 => "INT_24",
235                    ffi::PA_INT_16 => "INT_16",
236                    ffi::PA_INT_8 => "INT_8",
237                    ffi::PA_UINT_8 => "UINT_8",
238                    ffi::PA_CUSTOM_FORMAT => "CUSTOM_FORMAT",
239                    ffi::PA_NON_INTERLEAVED => "NON_INTERLEAVED",
240                    _ => "<Unknown SampleFormatFlags>",
241                }
242            )
243        }
244    }
245}

283/// A structure containing information about a particular host API.
284#[derive(Clone, Debug, PartialEq)]
285pub struct HostApiInfo<'a> {
286    /// The version of the struct
287    pub struct_version: i32,
288    /// The type of the current host
289    pub host_type: HostApiTypeId,
290    /// The name of the host
291    pub name: &'a str,
292    /// The total count of device in the host
293    pub device_count: u32,
294    /// The index to the default input device or None if no input device is available
295    pub default_input_device: Option<DeviceIndex>,
296    /// The index to the default output device or None if no output device is available
297    pub default_output_device: Option<DeviceIndex>,
298}
299
300impl<'a> HostApiInfo<'a> {
301    /// Construct the HostApiInfo from the equivalent C struct.
302    ///
303    /// Returns `None` if:
304    /// - either of the given device indices are invalid.
305    /// - the device_count is less than `0`.
306    /// - a valid `HostApiTypeId` can't be constructed from the given `host_type`.
307    pub fn from_c_info(c_info: ffi::PaHostApiInfo) -> Option<HostApiInfo<'a>> {
308        let default_input_device = match c_info.defaultInputDevice {
309            idx if idx >= 0 => Some(DeviceIndex(idx as u32)),
310            ffi::PA_NO_DEVICE => None,
311            _ => return None,
312        };
313        let default_output_device = match c_info.defaultOutputDevice {
314            idx if idx >= 0 => Some(DeviceIndex(idx as u32)),
315            ffi::PA_NO_DEVICE => None,
316            _ => return None,
317        };
318        let device_count = match c_info.deviceCount {
319            n if n >= 0 => n as u32,
320            _ => return None,
321        };
322        let host_type = match FromPrimitive::from_u32(c_info.type_) {
323            Some(ty) => ty,
324            None => return None,
325        };
326        Some(HostApiInfo {
327            struct_version: c_info.structVersion,
328            host_type: host_type,
329            name: ffi::c_str_to_str(c_info.name).unwrap_or("<Failed to convert str from CStr>"),
330            device_count: device_count,
331            default_input_device: default_input_device,
332            default_output_device: default_output_device,
333        })
334    }
335}
336
337impl<'a> From<HostApiInfo<'a>> for ffi::PaHostApiInfo {
338    fn from(info: HostApiInfo<'a>) -> Self {
339        let default_input_device = match info.default_input_device {
340            Some(i) => i.into(),
341            None => ffi::PA_NO_DEVICE,
342        };
343        let default_output_device = match info.default_output_device {
344            Some(i) => i.into(),
345            None => ffi::PA_NO_DEVICE,
346        };
347        ffi::PaHostApiInfo {
348            structVersion: info.struct_version as raw::c_int,
349            type_: info.host_type as ffi::PaHostApiTypeId,
350            name: ffi::str_to_c_str(info.name),
351            deviceCount: info.device_count as raw::c_int,
352            defaultInputDevice: default_input_device,
353            defaultOutputDevice: default_output_device,
354        }
355    }
356}

358/// Structure used to return information about a host error condition.
359#[derive(Clone, PartialEq, PartialOrd, Debug)]
360pub struct HostErrorInfo<'a> {
361    /// The host API which returned the error code
362    pub host_api_type: HostApiTypeId,
363    /// The code of the error
364    pub code: u32,
365    /// The string which explain the error
366    pub text: &'a str,
367}
368
369impl<'a> HostErrorInfo<'a> {
370    /// Construct a HostErrorInfo from the equivalent C struct.
371    pub fn from_c_error_info(c_error: ffi::PaHostErrorInfo) -> HostErrorInfo<'a> {
372        HostErrorInfo {
373            host_api_type: FromPrimitive::from_u32(c_error.hostApiType).unwrap(),
374            code: c_error.errorCode as u32,
375            text: ffi::c_str_to_str(c_error.errorText)
376                .unwrap_or("<Failed to convert str from CStr>"),
377        }
378    }
379}
380
381impl<'a> From<HostErrorInfo<'a>> for ffi::PaHostErrorInfo {
382    fn from(error: HostErrorInfo<'a>) -> Self {
383        ffi::PaHostErrorInfo {
384            hostApiType: FromPrimitive::from_i32(error.host_api_type as i32).unwrap(),
385            errorCode: error.code as raw::c_long,
386            errorText: ffi::str_to_c_str(error.text),
387        }
388    }
389}

391/// A structure providing information and capabilities of PortAudio devices.
392///
393/// Devices may support input, output or both input and output.
394#[derive(Clone, PartialEq, PartialOrd, Debug)]
395pub struct DeviceInfo<'a> {
396    /// The version of the struct
397    pub struct_version: i32,
398    /// The name of the device
399    pub name: &'a str,
400    /// Host API identifier
401    pub host_api: HostApiIndex,
402    /// Maximal number of input channels for this device
403    pub max_input_channels: i32,
404    /// maximal number of output channel for this device
405    pub max_output_channels: i32,
406    /// The default low latency for input with this device
407    pub default_low_input_latency: Time,
408    /// The default low latency for output with this device
409    pub default_low_output_latency: Time,
410    /// The default high latency for input with this device
411    pub default_high_input_latency: Time,
412    /// The default high latency for output with this device
413    pub default_high_output_latency: Time,
414    /// The default sample rate for this device
415    pub default_sample_rate: f64,
416}
417
418impl<'a> DeviceInfo<'a> {
419    /// Construct a **DeviceInfo** from the equivalent C struct.
420    pub fn from_c_info(c_info: ffi::PaDeviceInfo) -> DeviceInfo<'a> {
421        DeviceInfo {
422            struct_version: c_info.structVersion,
423            name: ffi::c_str_to_str(c_info.name).unwrap_or("<Failed to convert str from CStr>"),
424            host_api: c_info.hostApi,
425            max_input_channels: c_info.maxInputChannels,
426            max_output_channels: c_info.maxOutputChannels,
427            default_low_input_latency: c_info.defaultLowInputLatency,
428            default_low_output_latency: c_info.defaultLowOutputLatency,
429            default_high_input_latency: c_info.defaultHighInputLatency,
430            default_high_output_latency: c_info.defaultHighOutputLatency,
431            default_sample_rate: c_info.defaultSampleRate,
432        }
433    }
434}
