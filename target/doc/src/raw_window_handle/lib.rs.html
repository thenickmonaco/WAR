4//! Interoperability library for Rust Windowing applications.
5//!
6//! This library provides standard types for accessing a window's platform-specific raw window
7//! handle. This does not provide any utilities for creating and managing windows; instead, it
8//! provides a common interface that window creation libraries (e.g. Winit, SDL) can use to easily
9//! talk with graphics libraries (e.g. gfx-hal).
10//!
11//! ## Safety guarantees
12//!
13//! Please see the docs of [`HasRawWindowHandle`].
14//!
15//! ## Platform handle initialization
16//!
17//! Each platform handle struct is purposefully non-exhaustive, so that additional fields may be
18//! added without breaking backwards compatibility. Each struct provides an `empty` method that may
19//! be used along with the struct update syntax to construct it. See each specific struct for
20//! examples.

43/// Window that wraps around a raw window handle.
44///
45/// # Safety guarantees
46///
47/// Users can safely assume that non-`null`/`0` fields are valid handles, and it is up to the
48/// implementer of this trait to ensure that condition is upheld.
49///
50/// Despite that qualification, implementers should still make a best-effort attempt to fill in all
51/// available fields. If an implementation doesn't, and a downstream user needs the field, it should
52/// try to derive the field from other fields the implementer *does* provide via whatever methods the
53/// platform provides.
54///
55/// The exact handles returned by `raw_window_handle` must remain consistent between multiple calls
56/// to `raw_window_handle` as long as not indicated otherwise by platform specific events.
57pub unsafe trait HasRawWindowHandle {
58    fn raw_window_handle(&self) -> RawWindowHandle;
59}
60
61unsafe impl<'a, T: HasRawWindowHandle + ?Sized> HasRawWindowHandle for &'a T {
62    fn raw_window_handle(&self) -> RawWindowHandle {
63        (*self).raw_window_handle()
64    }
65}
66#[cfg(feature = "alloc")]
67#[cfg_attr(docsrs, doc(cfg(feature = "alloc")))]
68unsafe impl<T: HasRawWindowHandle + ?Sized> HasRawWindowHandle for alloc::rc::Rc<T> {
69    fn raw_window_handle(&self) -> RawWindowHandle {
70        (**self).raw_window_handle()
71    }
72}
73#[cfg(feature = "alloc")]
74#[cfg_attr(docsrs, doc(cfg(feature = "alloc")))]
75unsafe impl<T: HasRawWindowHandle + ?Sized> HasRawWindowHandle for alloc::sync::Arc<T> {
76    fn raw_window_handle(&self) -> RawWindowHandle {
77        (**self).raw_window_handle()
78    }
79}

81/// An enum to simply combine the different possible raw window handle variants.
82///
83/// # Variant Availability
84///
85/// Note that all variants are present on all targets (none are disabled behind
86/// `#[cfg]`s), but see the "Availability Hints" section on each variant for
87/// some hints on where this variant might be expected.
88///
89/// Note that these "Availability Hints" are not normative. That is to say, a
90/// [`HasRawWindowHandle`] implementor is completely allowed to return something
91/// unexpected. (For example, it's legal for someone to return a
92/// [`RawWindowHandle::Xlib`] on macOS, it would just be weird, and probably
93/// requires something like XQuartz be used).
94#[non_exhaustive]
95#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
96pub enum RawWindowHandle {
97    /// A raw window handle for UIKit (Apple's non-macOS windowing library).
98    ///
99    /// ## Availability Hints
100    /// This variant is likely to be used on iOS, tvOS, (in theory) watchOS, and
101    /// Mac Catalyst (`$arch-apple-ios-macabi` targets, which can notably use
102    /// UIKit *or* AppKit), as these are the targets that (currently) support
103    /// UIKit.
104    UiKit(UiKitHandle),
105    /// A raw window handle for AppKit.
106    ///
107    /// ## Availability Hints
108    /// This variant is likely to be used on macOS, although Mac Catalyst
109    /// (`$arch-apple-ios-macabi` targets, which can notably use UIKit *or*
110    /// AppKit) can also use it despite being `target_os = "ios"`.
111    AppKit(AppKitHandle),
112    /// A raw window handle for the Redox operating system.
113    ///
114    /// ## Availability Hints
115    /// This variant is used by the Orbital Windowing System in the Redox
116    /// operating system.
117    Orbital(OrbitalHandle),
118    /// A raw window handle for Xlib.
119    ///
120    /// ## Availability Hints
121    /// This variant is likely to show up anywhere someone manages to get X11
122    /// working that Xlib can be built for, which is to say, most (but not all)
123    /// Unix systems.
124    Xlib(XlibHandle),
125    /// A raw window handle for Xcb.
126    ///
127    /// ## Availability Hints
128    /// This variant is likely to show up anywhere someone manages to get X11
129    /// working that XCB can be built for, which is to say, most (but not all)
130    /// Unix systems.
131    Xcb(XcbHandle),
132    /// A raw window handle for Wayland.
133    ///
134    /// ## Availability Hints
135    /// This variant should be expected anywhere Wayland works, which is
136    /// currently some subset of unix systems.
137    Wayland(WaylandHandle),
138    /// A raw window handle for Win32.
139    ///
140    /// ## Availability Hints
141    /// This variant is used on Windows systems.
142    Win32(Win32Handle),
143    /// A raw window handle for WinRT.
144    ///
145    /// ## Availability Hints
146    /// This variant is used on Windows systems.
147    WinRt(WinRtHandle),
148    /// A raw window handle for the Web.
149    ///
150    /// ## Availability Hints
151    /// This variant is used on Wasm or asm.js targets when targeting the Web/HTML5.
152    Web(WebHandle),
153    /// A raw window handle for Android NDK.
154    ///
155    /// ## Availability Hints
156    /// This variant is used on Android targets.
157    AndroidNdk(AndroidNdkHandle),
158    /// A raw window handle for Haiku.
159    ///
160    /// ## Availability Hints
161    /// This variant is used on HaikuOS.
162    Haiku(HaikuHandle),
163}
