3use glfw::{Context, WindowHint, WindowMode};
4use glow::HasContext;
5
6pub fn render() {
7    let mut glfw: glfw::Glfw =
8        glfw::init(glfw::FAIL_ON_ERRORS).expect("Failed to init glfw");
9
10    glfw.window_hint(WindowHint::ContextVersionMajor(3));
11    glfw.window_hint(WindowHint::ContextVersionMinor(3));
12    glfw.window_hint(WindowHint::OpenGlProfile(glfw::OpenGlProfileHint::Core));
13
14    let (mut window, _events): (
15        glfw::Window,
16        std::sync::mpsc::Receiver<(f64, glfw::WindowEvent)>,
17    ) = glfw
18        .create_window(1920, 1080, "vimDAW", WindowMode::Windowed)
19        .expect("Failed to create GLFW window");
20
21    window.make_current();
22    window.set_key_polling(true);
23
24    let gl = unsafe {
25        glow::Context::from_loader_function(|s| {
26            window.get_proc_address(s) as *const _
27        })
28    };
29
30    while !window.should_close() {
31        unsafe {
32            gl.clear_color(0.0, 0.0, 0.0, 0.0);
33            gl.clear(glow::COLOR_BUFFER_BIT);
34        }
35
36        window.swap_buffers();
37        glfw.poll_events();
38    }
39}
