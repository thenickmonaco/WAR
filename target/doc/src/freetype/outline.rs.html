49const TAG_ONCURVE: c_char = 0x01;
50const TAG_BEZIER3: c_char = 0x02;
51
52pub struct CurveIterator<'a> {
53    start_point: *const Vector,
54    start_tag: *const c_char,
55    idx: isize,
56    length: isize,
57    marker: PhantomData<&'a ()>
58}
59
60impl<'a> CurveIterator<'a> {
61    pub unsafe fn from_raw(outline: &'a ffi::FT_Outline,
62                               start_idx: isize,
63                               end_idx: isize) -> Self {
64        CurveIterator {
65            start_point: outline.points.offset(start_idx),
66            start_tag: outline.tags.offset(start_idx),
67            idx: 0,
68            length: end_idx - start_idx + 1,
69            marker: PhantomData
70        }
71    }
72
73    pub fn start(&self) -> &'a Vector {
74        unsafe {
75            &*self.start_point
76        }
77    }
78
79    // Retrieves the point at offset i from the current point. Note that contours implicitly repeat their
80    // first point at the end.
81    unsafe fn pt(&self, i: isize) -> Vector {
82        if self.idx + i < self.length {
83            *self.start_point.offset(self.idx + i)
84        } else {
85            *self.start_point
86        }
87    }
88
89    unsafe fn tg(&self, i: isize) -> c_char {
90        if self.idx + i < self.length {
91            *self.start_tag.offset(self.idx + i)
92        } else {
93            *self.start_tag
94        }
95    }
96}
97
98impl<'a> Iterator for CurveIterator<'a> {
99    type Item = Curve;
100
101    fn next(&mut self) -> Option<Self::Item> {
102        if self.idx >= self.length {
103            None
104        } else {
105            unsafe {
106                let tag1 = self.tg(1);
107
108                let (shift, curve) = if (tag1 & TAG_ONCURVE) == TAG_ONCURVE {
109                    (1, Curve::Line(self.pt(1)))
110                } else if (tag1 & TAG_BEZIER3) == TAG_BEZIER3 {
111                    (3, Curve::Bezier3(self.pt(1), self.pt(2), self.pt(3)))
112                } else {
113                    // We are some kind of quadratic Bezier.
114                    // Quadratic Bezier curves have a special treatment in TTF outlines:
115                    // as an optimization, curves are often constructed from sequences
116                    // of off-curve control points. In this case, there are implied on-curve
117                    // points in between each pair of off-curve points.
118                    if (self.tg(2) & TAG_ONCURVE) == TAG_ONCURVE {
119                        (2, Curve::Bezier2(self.pt(1), self.pt(2)))
120                    } else {
121                        let pt = ffi::FT_Vector {
122                            x: (self.pt(1).x + self.pt(2).x) / 2,
123                            y: (self.pt(1).y + self.pt(2).y) / 2,
124                        };
125
126                        (1, Curve::Bezier2(self.pt(1), pt))
127                    }
128                };
129
130                self.idx += shift;
131                Some(curve)
132            }
133        }
134    }
135}

137pub struct ContourIterator<'a> {
138    outline: &'a ffi::FT_Outline,
139    contour_start: c_short,
140    contour_end_idx: *const c_short,
141    last_end_idx: *const c_short
142}
143
144impl<'a> ContourIterator<'a> {
145    pub unsafe fn from_raw(outline: &'a ffi::FT_Outline) -> Self {
146        ContourIterator {
147            outline: outline,
148            contour_start: 0,
149            contour_end_idx: outline.contours,
150            last_end_idx: outline.contours.offset(outline.n_contours as isize - 1)
151        }
152    }
153}
154
155impl<'a> Iterator for ContourIterator<'a> {
156    type Item = CurveIterator<'a>;
157
158    fn next(&mut self) -> Option<Self::Item> {
159        if self.contour_end_idx > self.last_end_idx {
160            None
161        } else {
162            unsafe {
163                let contour_end = *self.contour_end_idx;
164                let curves = CurveIterator::from_raw(self.outline, self.contour_start as isize,
165                                                     contour_end as isize);
166                self.contour_start = contour_end + 1;
167                self.contour_end_idx = self.contour_end_idx.offset(1);
168
169                Some(curves)
170            }
171        }
172    }
173}
