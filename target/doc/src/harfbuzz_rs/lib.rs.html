1//! `harfbuzz_rs` is a high-level interface to HarfBuzz, exposing its most important functionality
2//! in a safe manner using Rust.
3//!
4//! # What is HarfBuzz?
5//! HarfBuzz is a library for performing complex text layout. It does not perform any drawing. This
6//! is quite a low-level operation. If you want to simply draw some text on the screen choose
7//! another library. However if you want to build a library for drawing text on some canvas or
8//! need a lot of control on advanced text layout then this is the right library to use.
9//!
10//! # Getting Started
11//!
12//! To shape a simple string of text you just create a `Font` from a font file, fill a `Buffer`
13//! with some text and call the `shape` function.
14//!
15//! ```
16//! use harfbuzz_rs::*;
17//!
18//! # fn try_main() -> Result<(), std::io::Error> {
19//!
20//! let path = "path/to/some/font_file.otf";
21//! let index = 0; //< face index in the font file
22//! # let path = "testfiles/SourceSansVariable-Roman.ttf";
23//! let face = Face::from_file(path, index)?;
24//! let mut font = Font::new(face);
25//!
26//! let buffer = UnicodeBuffer::new().add_str("Hello World!");
27//! let output = shape(&font, buffer, &[]);
28//!
29//! // The results of the shaping operation are stored in the `output` buffer.
30//!
31//! let positions = output.get_glyph_positions();
32//! let infos = output.get_glyph_infos();
33//!
34//! # assert_eq!(positions.len(), 12);
35//! assert_eq!(positions.len(), infos.len());
36//!
37//! // iterate over the shaped glyphs
38//! for (position, info) in positions.iter().zip(infos) {
39//!     let gid = info.codepoint;
40//!     let cluster = info.cluster;
41//!     let x_advance = position.x_advance;
42//!     let x_offset = position.x_offset;
43//!     let y_offset = position.y_offset;
44//!
45//!     // Here you would usually draw the glyphs.
46//!     println!("gid{:?}={:?}@{:?},{:?}+{:?}", gid, cluster, x_advance, x_offset, y_offset);
47//! }
48//!
49//! # Ok(())
50//! # }
51//! #
52//! # try_main().unwrap();
53//! ```
54//! This should print out something similar to the following:
55//!
56//! ```text
57//! gid41=0@741,0+0
58//! gid70=1@421,0+0
59//! gid77=2@258,0+0
60//! gid77=3@253,0+0
61//! gid80=4@510,0+0
62//! gid1=5@227,0+0
63//! gid56=6@874,0+0
64//! gid80=7@498,0+0
65//! gid83=8@367,0+0
66//! gid77=9@253,0+0
67//! gid69=10@528,0+0
68//! gid2=11@276,0+0
69//! ```

99pub(crate) fn start_end_range(range: impl RangeBounds<usize>) -> (c_uint, c_uint) {
100    // We have to do careful bounds checking since c_uint may be of
101    // different sizes on different platforms. We do assume that
102    // sizeof(usize) >= sizeof(c_uint).
103    const MAX_UINT: usize = c_uint::max_value() as usize;
104    let start = match range.start_bound() {
105        Bound::Included(&included) => included.min(MAX_UINT) as c_uint,
106        Bound::Excluded(&excluded) => excluded.min(MAX_UINT - 1) as c_uint + 1,
107        Bound::Unbounded => 0,
108    };
109    let end = match range.end_bound() {
110        Bound::Included(&included) => included.saturating_add(1).min(MAX_UINT) as c_uint,
111        Bound::Excluded(&excluded) => excluded.min(MAX_UINT) as c_uint,
112        Bound::Unbounded => c_uint::max_value(),
113    };
114    (start, end)
115}

117/// A variation selector which can be applied to a specific font.
118///
119/// To use OpenType variations when shaping see the documentation of [`Font`].
120///
121/// # Fields
122///
123/// - `tag`: The OpenType tag of the variation.
124/// - `value`: Some OpenType variant accept different values to change
125///   their behaviour.
126#[derive(Debug, Copy, Clone)]
127#[repr(transparent)]
128pub struct Variation(hb::hb_variation_t);
129
130impl Variation {
131    /// Create a new Variation with provided `tag` and `value`.
132    ///
133    /// # Examples
134    ///
135    /// ```
136    /// use harfbuzz_rs::Variation;
137    /// Variation::new(b"wght", 800.0);
138    /// ```
139    pub fn new(tag: impl Into<Tag>, value: f32) -> Variation {
140        Variation(hb::hb_variation_t {
141            tag: tag.into().0,
142            value,
143        })
144    }
145
146    /// Returns the `tag` of the variation.
147    pub fn tag(&self) -> Tag {
148        Tag(self.0.tag)
149    }
150
151    /// Returns the value of the variation.
152    pub fn value(&self) -> f32 {
153        self.0.value
154    }
155}

157/// A feature tag with an accompanying range specifying on which subslice of
158/// `shape`s input it should be applied.
159///
160/// You can pass a slice of `Feature`s to `shape` that will be activated for the
161/// corresponding slices of input.
162///
163/// # Examples
164///
165/// Shape some text using the `calt` (Contextual Alternatives) feature.
166///
167/// ```
168/// use harfbuzz_rs::{Face, Font, UnicodeBuffer, shape, Feature, Tag};
169///
170/// let path = "testfiles/SourceSansVariable-Roman.ttf";
171/// let face = Face::from_file(path, 0).expect("could not load face");
172/// let font = Font::new(face);
173///
174/// let buffer = UnicodeBuffer::new().add_str("Hello World!");
175///
176/// // contextual alternatives feature
177/// let feature_tag = b"calt";
178///
179/// // use the feature on the entire input
180/// let feature_range = 0..;
181/// let feature = Feature::new(feature_tag, 0, feature_range);
182///
183/// let output = shape(&font, buffer, &[feature]);
184/// ```
185#[derive(Debug, Copy, Clone)]
186#[repr(transparent)]
187pub struct Feature(hb::hb_feature_t);
188
189impl Feature {
190    /// Create a new `Feature` struct.
191    ///
192    /// The feature will be applied with the given value to all glyphs which are
193    /// in clusters contained in `range`.
194    ///
195    /// # Arguments
196    ///
197    /// - `tag`: The OpenType feature tag to use.
198    /// - `value`: Some OpenType features accept different values to change
199    ///   their behaviour.
200    /// - `range`: The cluster range that should be affected by this feature.
201    pub fn new(tag: impl Into<Tag>, value: u32, range: impl RangeBounds<usize>) -> Feature {
202        let (start, end) = start_end_range(range);
203        Feature(hb::hb_feature_t {
204            tag: tag.into().0,
205            value,
206            start,
207            end,
208        })
209    }
210
211    pub fn tag(&self) -> Tag {
212        Tag(self.0.tag)
213    }
214
215    pub fn value(&self) -> u32 {
216        self.0.value
217    }
218
219    pub fn start(&self) -> usize {
220        self.0.start as usize
221    }
222
223    pub fn end(&self) -> usize {
224        self.0.end as usize
225    }
226}

228/// Shape the contents of the buffer using the provided font and activating all
229/// OpenType features given in `features`.
230///
231/// This function consumes the `buffer` and returns a `GlyphBuffer` containing
232/// the resulting glyph indices and the corresponding positioning information.
233/// Once all the information from the `GlyphBuffer` has been processed as
234/// necessary you can reuse the `GlyphBuffer` as an `UnicodeBuffer` (using
235/// `GlyphBuffer::clear_contents`) and use that to call `shape` again with new
236/// data.
237///
238/// By default some basic OpenType features are enabled according to the
239/// language and the script set in the buffer.
240///
241/// # Arguments
242/// - `font` – a reference to the harfbuzz font used to shape the text.
243/// - `buffer` – a `UnicodeBuffer` that is filled with the text to be shaped and
244/// also contains metadata about the text in the form of segment properties.
245/// - `features` – a slice of additional features to activate
246pub fn shape(font: &Font<'_>, buffer: UnicodeBuffer, features: &[Feature]) -> GlyphBuffer {
247    let buffer = buffer.guess_segment_properties();
248    unsafe {
249        hb::hb_shape(
250            font.as_raw(),
251            buffer.0.as_raw(),
252            features.as_ptr() as *mut _,
253            features.len() as u32,
254        )
255    };
256    GlyphBuffer(buffer.0)
257}
